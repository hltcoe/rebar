76      public static void writeLid(Corpus corpus, Set<Stage> deps, String version) throws RebarException {
77          // Our output field:
78          final FieldDescriptor lidField = Concrete.Communication.getDescriptor().findFieldByName("language_id");
79  
80          logger.info("Writing...");
81          final Stage lidStage = corpus.makeStage("lid", version, deps, "test stage", true);
82          final Corpus.Reader reader = corpus.makeReader(deps);
83          final Corpus.Writer lidWriter = corpus.makeWriter(lidStage);
84          final Iterator<IndexedCommunication> readIter = reader.loadCommunications();
85          while (readIter.hasNext()) {
86              IndexedCommunication com = readIter.next();
87              Concrete.LanguageIdentification.Builder lidBuilder = Concrete.LanguageIdentification.newBuilder();
88              Concrete.LanguageIdentification.LanguageProb.Builder lp = Concrete.LanguageIdentification.LanguageProb.newBuilder();
89              lp.setProbability(1.0f);
90              lp.setLanguage("eng");
91              lidBuilder.addLanguage(lp);
92              lidBuilder.setUuid(IdUtil.generateUUID());
93              lidBuilder.getMetadataBuilder().setTool("my-lid");
94              com.addField(com.getProto(), lidField, lidBuilder.build());
95              lidWriter.saveCommunication(com);
96          }
97          reader.close();
98          lidWriter.close();
99      }

101     public static void read(Corpus corpus, Collection<Stage> stages) throws RebarException {
102         Collection<String> subset = Arrays.asList(new String[] { "doc-3", "doc-9" });
103         corpus.registerComIdSet("my-ids", subset);
104 
105         logger.info("Reading...");
106         Corpus.Reader reader = corpus.makeReader(stages);
107         subset = corpus.lookupComIdSet("all");
108         for (String s : subset)
109             logger.info("HELLO " + s);
110         Iterator<IndexedCommunication> readIter = reader.loadCommunications(subset);
111         while (readIter.hasNext()) {
112             IndexedCommunication com = readIter.next();
113             logger.info("===================\n" + com);
114         }
115         reader.close();
116         logger.info("All done");
117     }

119     public static void initialize(Corpus corpus) throws RebarException {
120         Corpus.Initializer initializer = corpus.makeInitializer();
121         for (int i = 0; i < 10; ++i) {
122             String docid = "doc-" + i;
123             String text = "this is document " + i + ".";
124             Concrete.CommunicationGUID guid = Concrete.CommunicationGUID.newBuilder().setCommunicationId(docid)
125                     .setCorpusName(corpus.getName()).build();
126             Concrete.Communication com = Concrete.Communication.newBuilder().setText(text).setUuid(IdUtil.generateUUID()).setGuid(guid)
127                     .setKnowledgeGraph(Concrete.KnowledgeGraph.newBuilder().setUuid(IdUtil.generateUUID()).build()).build();
128             initializer.addCommunication(com);
129         }
130         initializer.close();
131     }
