19  /**
20   * Interface for reading and writing from a "REBAR Corpus".  Each
21   * "REBAR Corpus" consists of a set of Communications.
22   *
23   * Each corpus is backed by some external shared resource (such as an
24   * Accumulo table or a directory on a filesystem), and any changes
25   * made to the corpus are persistant.
26   *
27   * Corpus objects are created using the factory class
28   * "Corpus.Factory".  In particular, use Corpus.Factory.makeCorpus()
29   * to create a new corpus, and Corpus.Factory.getCorpus() to get a
30   * connection to an existing corpus.
31   */
32  public interface Corpus extends StagedDataCollection {
33  
34      // ======================================================================
35      // Identifier Sets
36      // ======================================================================
37  
38      /** Return a new identifier set containing the identifiers listed
39       * in the specified file.  The file should contain one identifier
40       * per line. */
41      public Collection<String> readComIdSet(File filename) throws RebarException;
42  
43      /** Permanently associate a name with a given identifier set.
44       * This identifier set can then be looked up using the method
45       * 'lookupComIdSet()'. */
46      public void registerComIdSet(String name, Collection<String> idSet) throws RebarException;
47  
48      /** Look up and return a symbolic identifier set that has been
49       * defined using the 'registerComIdSet()' method. */
50      public Collection<String> lookupComIdSet(String name) throws RebarException;
51  
52      /** Get a list of the names of all symbolic communication
53       * identifier sets that have been defined for this corpus. */
54      public Collection<String> getComIdSetNames() throws RebarException;
55  
56      // ======================================================================
57      // Initializer
58      // ======================================================================
59  
60      /** Interface used to initialize a new corpus by defining the set
61       * of communications that it contains.  
62       *
63       * (This interface could also be used to add new Communications to
64       * an existing corpus, but this should be done with care, since
65       * any stages that have been run will not contain output for these
66       * new Communications.)
67       */
68      interface Initializer {
69  	public IndexedCommunication addCommunication(Concrete.Communication comm) throws RebarException;
70  	public boolean communicationExists(String commId) throws RebarException;
71  	public void close() throws RebarException;
72      }
73  
74      /** Return a Corpus.Initializer that can be used to add
75       * Communication objects to this corpus. */
76      public Initializer makeInitializer() throws RebarException;

78      // ======================================================================
79      // Reader

81  
82      /** Interface for "corpus readers", which are used to read a
83       * communications from a corpus, including the output of a
84       * specified set of stages. */
85      interface Reader {
86  	/** Read a single communication */
87  	public IndexedCommunication loadCommunication(String comid) throws RebarException;
88  
89  	/** Read a specified set of communications.
90  	 *
91  	 * @param subset A collection identifying which set of
92  	 *     communications should be returned by the iterator.
93  	 **/
94  	public Iterator<IndexedCommunication> loadCommunications(Collection<String> subset) throws RebarException;
95  
96  	/** Read all communications */
97  	public Iterator<IndexedCommunication> loadCommunications() throws RebarException;
98  
99  	/** Release any resources associated with this reader. */
100 	public void close() throws RebarException;
101 
102 	/** Return the list of input stages for this reader. */
103 	public Collection<Stage> getInputStages() throws RebarException;
104 		
105 	/** Return the corpus that owns this reader. */
106 	Corpus getCorpus();
107     }

118     public Reader makeReader() throws RebarException;

121     public Reader makeReader(Stage stages[], boolean loadStageOwnership) throws RebarException;

132     /** Interface for "corpus writers," which are used to add
133      * the output of a processing stage to some or all of the
134      * communications in a corpus.
135      */
136     interface Writer {
137 	/** Save any changes that have been made to the given
138 	 * communication (or do nothing if no changes have been made).
139 	 * Changes can be made to an IndexedCommunication using its
140 	 * addField() and SetField() methods, as well as various
141 	 * convenience methods (such as addTokenization()) that
142 	 * delegate to these methods. */
143 	public void saveCommunication(IndexedCommunication comm) throws RebarException;
144 
145 	public void flush() throws RebarException;
146 	public void close() throws RebarException;
147 	public Stage getOutputStage();
148     }
149 
150     /** Return a Corpus.Writer that can be used to add the output
151      * of a processing stage to this corpus. */
152     public Writer makeWriter(Stage stage) throws RebarException;
153 
154     /** Interface used by RPC to write "diffs", i.e., monotonic
155      * changes to communications.  Each diff consists of a dictionary
156      * that maps from UUIDs of objects contained within a
157      * communication to serialized protobuf messages that should be
158      * merged into those objects.  Since diffs are merged in, they
159      * can only be used to make monotonic changes (i.e., add field
160      * values or set field values, not remove or modify them). */
161     interface DiffWriter {
162 	public void saveCommunicationDiff(String comid, Map<ProtoIndex.ModificationTarget, byte[]> changes) throws RebarException;
163 	public void flush() throws RebarException;
164 	public void close() throws RebarException;
165     }
166     public DiffWriter makeDiffWriter(Stage stage) throws RebarException;
