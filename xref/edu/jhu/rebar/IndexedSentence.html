17  public class IndexedSentence extends IndexedProto<Concrete.Sentence> {
18  	private final FieldDescriptor TOKENIZATION_FIELD =

21  	public static IndexedSentence build(Concrete.UUID uuid, ProtoIndex index) throws RebarException
22  	{
23  		IndexedSentence cached = index.getIndexedProto(uuid);
24  		if (cached != null) return cached;
25  		else return new IndexedSentence((Concrete.Sentence)index.lookup(uuid), index);
26  	}
27  
28  	public static IndexedSentence build(Concrete.Sentence sentence, ProtoIndex index) throws RebarException {
29  		IndexedSentence cached = index.getIndexedProto(sentence.getUuid());
30  		if (cached != null) return cached;
31  		else return new IndexedSentence(sentence, index);
32  
33  	}

35  	private IndexedSentence(Concrete.Sentence sentence, ProtoIndex index) throws RebarException
36  	{
37  		super(sentence, index);
38  	}

56  		return IndexedTokenization.build(getProto().getTokenization(index), getIndex());

59  	/** Return a list of IndexedTokenizations for the tokenizations in
60  	 * this sentence. */
61  	public List<IndexedTokenization> getTokenizationList() throws RebarException {
62  		List<IndexedTokenization> result = new ArrayList<IndexedTokenization>();
63  		for (Concrete.Tokenization tokz: getProto().getTokenizationList())
64  			result.add(IndexedTokenization.build(tokz, getIndex()));
65  		return result;
66  	}
67  
68  	/** Return an IndexedTokenization for the unique tokenization for
69  	 * this sentence.  If this sentence has no tokenization, or has
70  	 * multiple tokenizations, then throw an exception. */
71  	public IndexedTokenization getTokenization() throws RebarException {
72  		if (getProto().getTokenizationCount() == 1)
73  			return getTokenization(0);
74  		else if (getProto().getTokenizationCount() == 0)
75  			throw new RebarException("This sentence has no tokenization");
76  		else
77  			throw new RebarException("This sentence has multiple tokenizations");
78  	}
