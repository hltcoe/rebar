108     private void close() throws RebarException {
109         // Close everything we own.
110         for (Corpus.Initializer x : corpusInitializers.values())
111             x.close();
112         for (Corpus.Reader x : corpusReaders.values())
113             x.close();
114         for (Corpus.DiffWriter x : corpusDiffWriters.values())
115             x.close();
116         for (Graph.Initializer x : graphInitializers.values())
117             x.close();
118         for (Graph.Reader x : graphReaders.values())
119             x.close();
120         for (Graph.DiffWriter x : graphDiffWriters.values())
121             x.close();
122         for (Corpus c : corpora.values())
123             c.close();
124         for (Graph g : graphs.values())
125             g.close();
126         corpusInitializers.clear();
127         corpusReaders.clear();
128         corpusDiffWriters.clear();
129         graphInitializers.clear();
130         graphReaders.clear();
131         graphDiffWriters.clear();
132         corpora.clear();
133         graphs.clear();
134     }

136     public void run() throws IOException, RebarException {
137         while (true) {
138             // Read a varint (length) followed by the actual message.
139             RPCRequest request = RPCRequest.parseDelimitedFrom(inputStream);
140             try {
141                 if (request == null) {
142                     LOGGER.info("Stream disconnected!");
143                     return;
144                 }
145                 // Process the request
146                 RPCResponse response = handleRequest(request);
147                 // Send the response back
148                 response.writeDelimitedTo(outputStream);
149                 outputStream.flush();
150                 // Check whether we're finished.
151                 if (request.getClose())
152                     break;
153             } catch (RebarException e) {
154                 final StringWriter sw = new StringWriter();
155                 final PrintWriter pw = new PrintWriter(sw);
156                 e.printStackTrace(pw);
157                 buildErrorResponse(e.toString() + "\n" + sw.toString()).writeDelimitedTo(outputStream);
158                 outputStream.flush();
159             }
160         }
161     }

172         else if (request.hasVertexIteratorNext())
173             return handleVertexIteratorNext(request.getVertexIteratorNext());
174         else if (request.hasEdgeIteratorNext())
175             return handleEdgeIteratorNext(request.getEdgeIteratorNext());
176 
177         // 2. Iterator has-next methods
178         else if (request.hasCommunicationIteratorHasNext())
179             return handleCommunicationIteratorHasNext(request.getCommunicationIteratorHasNext());

191             return handleGraphDiffWriterWriteEdgeDiff(request.getGraphDiffWriterWriteEdgeDiff());
192

195             return handleGetCorpus(request.getGetCorpus());

347     private Map<ProtoIndex.ModificationTarget, byte[]> parseDiff(ConcreteRpc.ProtoDiff diff) {
348         final Map<ProtoIndex.ModificationTarget, byte[]> result = new HashMap<ProtoIndex.ModificationTarget, byte[]>();
349         for (ConcreteRpc.ProtoDiff.Entry entry : diff.getDiffList()) {
350             ProtoIndex.ModificationTarget target;
351             if (entry.hasUuid())
352                 target = new ProtoIndex.ModificationTarget(entry.getUuid());
353             else
354                 target = new ProtoIndex.ModificationTarget(entry.getEdgeId());
355             result.put(target, entry.getProto().toByteArray());
356         }
357         return result;
358     }

465     private ConcreteRpc.Stage buildRpcStage(Stage stage) {
466         List<Integer> dependencies = new ArrayList<Integer>();
467         for (Stage dep : stage.getDependencies())
468             dependencies.add(dep.getStageId());
469         return ConcreteRpc.Stage.newBuilder().setStageName(stage.getStageName()).setStageVersion(stage.getStageVersion())
470                 .setStageId(stage.getStageId()).addAllDependency(dependencies).setDescription(stage.getDescription())
471                 .setIsPublic(stage.isPublic()).build();
472     }

474     private void addStageOwnership(StageOwnership stageOwnership, RPCResponse.Builder response) {
475         if (stageOwnership == null)
476             return;
477         ConcreteRpc.StageOwnershipMap.Builder ownershipMap = response.addStageOwnershipBuilder();
478         for (Map.Entry<StageOwnership.FieldValuePointer, Stage> e : stageOwnership.entrySet()) {
479             StageOwnership.FieldValuePointer fvp = e.getKey();
480             List<ConcreteRpc.StageOwnershipMap.Segment> segments = new ArrayList<ConcreteRpc.StageOwnershipMap.Segment>();
481             while (fvp != null) {
482                 segments.add(ConcreteRpc.StageOwnershipMap.Segment.newBuilder().setFieldTag(fvp.field.getNumber()).setIndex(fvp.index)
483                         .build());
484                 fvp = fvp.parent;
485             }
486             Collections.reverse(segments);
487             ownershipMap.addEntryBuilder().setStageId(e.getValue().getStageId()).addAllPathSegment(segments);
488         }
489     }

528         RPCResponse.Builder builder = RPCResponse.newBuilder();

534     private RPCResponse handleCorpusMakeStage(ConcreteRpc.Corpus.MakeStage request) throws RebarException {
535         Corpus self = corpora.get(request.getSelf().getId());
536         Set<Stage> dependencies = new TreeSet<Stage>();
537         for (int dep : request.getDependencyList())
538             dependencies.add(self.getStage(dep));
539         Stage newStage = self.makeStage(request.getStageName(), request.getStageVersion(), dependencies, request.getDescription(),
540                 request.getDeleteIfExists());
541         return RPCResponse.newBuilder().addStage(buildRpcStage(newStage)).build();
542     }

697         Iterator<IndexedCommunication> self = communicationIterators.get(request.getSelf().getId());
698         int maxValues = request.getMaxValues();
699         int bytes = 0;
