47  /** 
48   * Processing class for reading a twitter corpus from a set of files,
49   * and writing the result to a Corpus.
50   *
51   * Adds the following Stages:
52   *   root -- contains guid, text, date
53   *   tweet_info -- contains all tweet info
54   *   default_section_segmentation -- contains a single section spanning the entire tweet
55   *   default_sentence_segmentation -- contains a single sentence spanning the entire tweet
56   *   com_graph -- initial communication graph, containing vertices for the
57   *      communication, the twitter account, and the sender.
58   */
59  public class TwitterCorpusIngester {

105     public TwitterCorpusIngester(Corpus corpus) throws RebarException {
106         Set<Stage> noDependencies = new TreeSet<Stage>();
107         this.corpus = corpus;
108         // For writing the root communication objects:
109         this.initializer = corpus.makeInitializer();
110         // Stage writer for tweet information (from twitter api)
111         if (!this.corpus.hasStage(TWEET_INFO_STAGE_NAME, TWEET_INFO_STAGE_VERSION)) {
112             this.tweetInfoStage = corpus.makeStage(TWEET_INFO_STAGE_NAME, TWEET_INFO_STAGE_VERSION, noDependencies,
113                     TWEET_INFO_STAGE_DESCRIPTION, false);
114         } else {
115             this.tweetInfoStage = this.corpus.getStage(TWEET_INFO_STAGE_NAME, TWEET_INFO_STAGE_VERSION);
116         }
117         this.tweetInfoWriter = corpus.makeWriter(this.tweetInfoStage);
118         // Stage writer for section segmentation
119         if (!this.corpus.hasStage(SEC_SEG_STAGE_NAME, SEC_SEG_STAGE_VERSION)) {
120             this.secSegStage = corpus
121                     .makeStage(SEC_SEG_STAGE_NAME, SEC_SEG_STAGE_VERSION, noDependencies, SEC_SEG_STAGE_DESCRIPTION, false);
122         } else {
123             this.secSegStage = this.corpus.getStage(SEC_SEG_STAGE_NAME, SEC_SEG_STAGE_VERSION);
124         }
125         this.secSegWriter = corpus.makeWriter(this.secSegStage);
126         // Stage writer for sentence segmentation
127         if (!this.corpus.hasStage(SENT_SEG_STAGE_NAME, SENT_SEG_STAGE_VERSION)) {
128             Set<Stage> sentSegDeps = new TreeSet<Stage>();
129             sentSegDeps.add(this.secSegStage);
130             this.sentSegStage = corpus.makeStage(SENT_SEG_STAGE_NAME, SENT_SEG_STAGE_VERSION, sentSegDeps, SENT_SEG_STAGE_DESCRIPTION,
131                     false);
132         } else {
133             this.sentSegStage = this.corpus.getStage(SENT_SEG_STAGE_NAME, SENT_SEG_STAGE_VERSION);
134         }
135         this.sentSegWriter = corpus.makeWriter(this.sentSegStage);
136         // Stage writer for initial knowledge graph
137         if (!this.corpus.hasStage(INITIAL_GRAPH_STAGE_NAME, INITIAL_GRAPH_STAGE_VERSION)) {
138             this.initialGraphStage = corpus.makeStage(INITIAL_GRAPH_STAGE_NAME, INITIAL_GRAPH_STAGE_VERSION, noDependencies,
139                     INITIAL_GRAPH_STAGE_DESCRIPTION, false);
140         } else {
141             this.initialGraphStage = this.corpus.getStage(INITIAL_GRAPH_STAGE_NAME, INITIAL_GRAPH_STAGE_VERSION);
142         }
143         this.initialGraphWriter = corpus.makeWriter(this.initialGraphStage);
144         // Metadata
145         this.attribMetadata = Concrete.AttributeMetadata.newBuilder().setTool("jhu.hltcoe.rebar2.ingest.TwitterCorpusIngester")
146                 .setConfidence(1.0f).build();
147         this.annotationMetadata = Concrete.AnnotationMetadata.newBuilder().setTool("jhu.hltcoe.rebar2.ingest.TwitterCorpusIngester")
148         // .setTimestamp()
149                 .build();
150         // Sanity checks
151         if (COM_GUID == null)
152             throw new RebarException("COM_GUID attribute not found");
153         if (TWITTER_ID == null)
154             throw new RebarException("TWITTER_ID attribute not found");
155         if (TWITTER_HANDLE == null)
156             throw new RebarException("TWITTER_HANDLE attribute not found");
157         if (IS_SENDER == null)
158             throw new RebarException("IS_SENDER attribute not found");
159         if (USES == null)
160             throw new RebarException("USES attribute not found");
161         if (COMS_SENT == null)
162             throw new RebarException("COMS_SENT attribute not found");
163     }

179     private static final DateTimeFormatter tweetDateFormat = DateTimeFormat.forPattern("EEE MMM dd HH:mm:ss Z YYYY");
180 
181     private IndexedCommunication addRoot(Concrete.TweetInfo tweet) throws RebarException {
182         // Use the tweet id as the document id.
183         String docid = new Long(tweet.getId()).toString();
184         Concrete.CommunicationGUID guid = Concrete.CommunicationGUID.newBuilder().setCommunicationId(docid).setCorpusName(corpus.getName())
185                 .build();
186         // Create an empty knowledge graph and assign it a uuid.
187         Concrete.KnowledgeGraph graph = Concrete.KnowledgeGraph.newBuilder().setUuid(IdUtil.generateUUID()).build();
188         // Create the basic communication with the tweet text
189         Concrete.Communication.Builder comBuilder = Concrete.Communication.newBuilder().setUuid(IdUtil.generateUUID()).setGuid(guid)
190                 .setText(tweet.getText()).setKind(Concrete.Communication.Kind.TWEET).setKnowledgeGraph(graph);
191         // Add the start time/date, if we have it.
192         if (tweet.hasCreatedAt()) {
193             try {
194                 DateTime dateTime = tweetDateFormat.parseDateTime(tweet.getCreatedAt());
195                 comBuilder.setStartTime(dateTime.getMillis() / 1000);
196             } catch (IllegalArgumentException iae) {
197                 logger.debug("Tweet " + docid + " had a malformed date. Skipping date creation.");
198             }
199         }
200         // Build it & add it to the corpus.
201         return initializer.addCommunication(comBuilder.build());
202     }

209     private void addSectionSegmentation(IndexedCommunication com) throws RebarException {
210         String text = com.getText();
211         // Wrap it in a simple segmentation.
212         Concrete.TextSpan textSpan = Concrete.TextSpan.newBuilder().setStart(0).setEnd(text.length()).build();
213         Concrete.SectionSegmentation.Builder segmentation = Concrete.SectionSegmentation.newBuilder();
214         segmentation.setUuid(IdUtil.generateUUID()).setMetadata(annotationMetadata).addSectionBuilder() // start
215                                                                                                         // new
216                                                                                                         // sub-builder
217                 .setUuid(IdUtil.generateUUID()).setTextSpan(textSpan);
218         com.addSectionSegmentation(segmentation.build());
219         secSegWriter.saveCommunication(com);
220     }
221 
222     private void addSentenceSegmentation(IndexedCommunication com) throws RebarException {
223         for (IndexedSection sec : com.getSections()) {
224             Concrete.SentenceSegmentation.Builder segmentation = Concrete.SentenceSegmentation.newBuilder();
225             segmentation.setUuid(IdUtil.generateUUID()).setMetadata(annotationMetadata).addSentenceBuilder() // start
226                                                                                                              // new
227                                                                                                              // sub-builder
228                     .setTextSpan(sec.getTextSpan()).setUuid(IdUtil.generateUUID());
229             sec.addSentenceSegmentation(segmentation.build());
230         }
231         sentSegWriter.saveCommunication(com);
232     }

245     private void addInitialGraph(IndexedCommunication com, Concrete.TweetInfo tweet) throws RebarException {
246         IndexedKnowledgeGraph graph = com.getKnowledgeGraph();
247 
248         // Create three vertices: one for the tweet, one for the
249         // person who sent it, and one for the account that was used
250         // to send it.
251         IndexedVertex senderVertex = graph.addVertex();
252         senderVertex.addAttribute(VERTEX_KIND, Concrete.Vertex.Kind.PERSON, attribMetadata);
253         IndexedVertex tweetVertex = graph.addVertex();
254         tweetVertex.addAttribute(VERTEX_KIND, Concrete.Vertex.Kind.COMMUNICATION, attribMetadata);
255         tweetVertex.addAttribute(COM_GUID, com.getGuid(), attribMetadata);
256         IndexedVertex accountVertex = graph.addVertex();
257         accountVertex.addAttribute(VERTEX_KIND, Concrete.Vertex.Kind.COM_CHANNEL, attribMetadata);
258         accountVertex.addAttribute(TWITTER_ID, tweet.getUser().getId(), attribMetadata);
259         accountVertex.addAttribute(TWITTER_HANDLE, tweet.getUser().getScreenName(), attribMetadata);
260 
261         // Edge between the sender and the message that they sent.
262         IndexedEdge senderTweetEdge = graph.getEdge(senderVertex, tweetVertex);
263         senderTweetEdge.addDirectedAttribute(senderVertex, IS_SENDER, tweetVertex, true, attribMetadata);
264 
265         // Edge between the sender and the account they used.
266         IndexedEdge senderAccountEdge = graph.getEdge(senderVertex, accountVertex);
267         senderAccountEdge.addDirectedAttribute(senderVertex, USES, accountVertex, true, attribMetadata);
268         senderAccountEdge.addDirectedAttribute(senderVertex, COMS_SENT, accountVertex, 1, attribMetadata);
269 
270         // Edge between the message and the account used to send it.
271         IndexedEdge accountTweetEdge = graph.getEdge(accountVertex, tweetVertex);
272         accountTweetEdge.addDirectedAttribute(accountVertex, IS_SENDER, tweetVertex, true, attribMetadata);
273 
274         // Save our changes.
275         initialGraphWriter.saveCommunication(com);
276     }

278     private void ingestTweet(Concrete.TweetInfo tweet) throws RebarException {
279         long tweetId = tweet.getId();
280         logger.debug("Tweet id: " + tweetId);
281 
282         if (dupeIds.contains(tweetId)) {
283             logger.debug("In our tweet ID set.");
284             this.dupes++;
285             return;
286         }
287 
288         else
289             logger.debug("Not in our tweet ID set.");
290 
291         if (this.initializer.communicationExists(Long.toString(tweetId))) {
292             logger.debug("Tweet " + tweetId + " has been ingested previously. Skipping.");
293             this.dupes++;
294             return;
295         }
296         // this.initialGraphWriter.
297 
298         IndexedCommunication com = addRoot(tweet);
299         addTweetInfo(com, tweet);
300         addSectionSegmentation(com);
301         addSentenceSegmentation(com);
302         addInitialGraph(com, tweet);
303         dupeIds.add(tweetId);
304         this.tweetsAdded++;
305     }

307     private void ingest(File filename) throws RebarException, IOException, FileNotFoundException {
308         InputStream is = FileUtil.getInputStream(filename);
309         Scanner sc = new Scanner(is, "UTF-8");
310         logger.info("Ingesting " + filename + "...");
311         long lineno = 0;
312         while (sc.hasNextLine()) {
313             try {
314                 String line = sc.nextLine();
315                 lineno++;
316                 Concrete.TweetInfo tweet = TweetInfoJsonReader.parseJson(line);
317                 if (tweet.hasText())
318                     ingestTweet(tweet);
319 
320                 if (lineno % 10000 == 0)
321                     logger.info("On line: " + lineno);
322             } catch (JsonParseException jpe) {
323                 logger.error("Couldn't parse this json: line " + lineno + ": " + jpe.getMessage());
324             } catch (JsonMappingException jme) {
325                 logger.error("Couldn't parse this json: line " + lineno + ": " + jme.getMessage());
326             }
327         }
328 
329         sc.close();
330         is.close();
331     }

341     public static void main(String[] args) throws RebarException, IOException, FileNotFoundException {
342         if (args.length < 2) {
343             logger.info("Usage: TwitterCorpusIngester <corpusname> <filename>...");
344             return;
345         }
346 
347         logger.info("Starting.");
348         // Thread.sleep(10000);
349         String corpusName = args[0];
350         CorpusFactory cf = RebarBackends.FILE.getCorpusFactory();
351         if (cf.corpusExists(corpusName)) {
352             logger.info("Deleting existing corpus: " + corpusName);
353             new FileCorpusFactory("target/").deleteCorpus(corpusName);
354         } else {
355             logger.error("Corpus " + corpusName + " does not exist. Creating it.");
356         }
357 
358         Corpus corpus = cf.makeCorpus(corpusName);
359         logger.info("Ingesting twitter files...");
360         TwitterCorpusIngester ingester = new TwitterCorpusIngester(corpus);
361         for (int i = 1; i < args.length; i++) {
362             ingester.ingest(new File(args[i]));
363         }
364         ingester.close();
365         logger.info("Closing corpus.");
366         logger.info("Tweets ingested: " + ingester.getTweetsAdded());
367         logger.info("Dupes: " + ingester.getDupes());
368         corpus.close();
369     }
