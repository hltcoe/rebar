25  /**
26   * Processing class for reading a twitter corpus from a set of files, and
27   * writing the result to a Corpus.
28   * 
29   * Adds the following Stages: root -- contains guid, text, date tweet_info --
30   * contains all tweet info default_section_segmentation -- contains a single
31   * section spanning the entire tweet default_sentence_segmentation -- contains a
32   * single sentence spanning the entire tweet com_graph -- initial communication
33   * graph, containing vertices for the communication, the twitter account, and
34   * the sender.
35   */
36  
37  public class TwitterCorpusMaker {
38      private static final Logger LOGGER = Logger.getLogger(TwitterCorpusMaker.class);
39  
40      private final Corpus corpus;
41      private final Corpus.Initializer initializer;
42      private final Concrete.AnnotationMetadata annotationMetadata;
43      private final Concrete.AttributeMetadata attribMetadata;
44  
45      // TweetInfo stage
46      private final FieldDescriptor TWEET_INFO_FIELD = Concrete.Communication.getDescriptor().findFieldByName("tweet_info");
47      private final String TWEET_INFO_STAGE_NAME = "tweet_info";
48      private final String TWEET_INFO_STAGE_VERSION = "1.0";
49      private final String TWEET_INFO_STAGE_DESCRIPTION = "Information about this tweet that is provided by the standard Twitter JSON API.";
50      private final Stage tweetInfoStage;
51  
52      // Section Segmentation Stage
53      private final String SEC_SEG_STAGE_NAME = "default_section_segmentation";
54      private final String SEC_SEG_STAGE_VERSION = "1.0";
55      private final String SEC_SEG_STAGE_DESCRIPTION = "A segmentation consisting of a single section, spanning the entire tweet.";
56      private final Stage secSegStage;
57  
58      // Section Segmentation Stage
59      private final String SENT_SEG_STAGE_NAME = "default_sentence_segmentation";
60      private final String SENT_SEG_STAGE_VERSION = "1.0";
61      private final String SENT_SEG_STAGE_DESCRIPTION = "A segmentation consisting of a single sentence, spanning the entire tweet.";
62      private final Stage sentSegStage;
63  
64      // InitialGraph stage
65      private final FieldDescriptor GRAPH_EDGE_FIELD = Concrete.KnowledgeGraph.getDescriptor().findFieldByName("vertex");
66      private final String INITIAL_GRAPH_STAGE_NAME = "com_graph";
67      private final String INITIAL_GRAPH_STAGE_VERSION = "1.0";
68      private final String INITIAL_GRAPH_STAGE_DESCRIPTION = "An initial communication graph, containing three vertices: one for the "
69              + "tweet (Communiation), one for the sender (Person), and one for the "
70              + "twitter account used to send the tweet (ComChannel).";
71      private final Stage initialGraphStage;
72  
73      private boolean duplicatesDetected = false;
74  
75      public TwitterCorpusMaker(Corpus corpus) throws RebarException {
76          Set<Stage> noDependencies = new TreeSet<Stage>();
77          this.corpus = corpus;
78          // For writing the root communication objects:
79          this.initializer = corpus.makeInitializer();
80          // Stage for tweet information (from twitter api)
81          if (!this.corpus.hasStage(TWEET_INFO_STAGE_NAME, TWEET_INFO_STAGE_VERSION)) {
82              this.tweetInfoStage = corpus.makeStage(TWEET_INFO_STAGE_NAME, TWEET_INFO_STAGE_VERSION, noDependencies,
83                      TWEET_INFO_STAGE_DESCRIPTION, false);
84          } else {
85              this.tweetInfoStage = this.corpus.getStage(TWEET_INFO_STAGE_NAME, TWEET_INFO_STAGE_VERSION);
86          }
87  
88          // Stage for section segmentation
89          if (!this.corpus.hasStage(SEC_SEG_STAGE_NAME, SEC_SEG_STAGE_VERSION)) {
90              this.secSegStage = corpus
91                      .makeStage(SEC_SEG_STAGE_NAME, SEC_SEG_STAGE_VERSION, noDependencies, SEC_SEG_STAGE_DESCRIPTION, false);
92          } else {
93              this.secSegStage = this.corpus.getStage(SEC_SEG_STAGE_NAME, SEC_SEG_STAGE_VERSION);
94          }
95  
96          // Stage for sentence segmentation
97          if (!this.corpus.hasStage(SENT_SEG_STAGE_NAME, SENT_SEG_STAGE_VERSION)) {
98              Set<Stage> sentSegDeps = new TreeSet<Stage>();
99              sentSegDeps.add(this.secSegStage);
100             this.sentSegStage = corpus.makeStage(SENT_SEG_STAGE_NAME, SENT_SEG_STAGE_VERSION, sentSegDeps, SENT_SEG_STAGE_DESCRIPTION,
101                     false);
102         } else {
103             this.sentSegStage = this.corpus.getStage(SENT_SEG_STAGE_NAME, SENT_SEG_STAGE_VERSION);
104         }
105 
106         // Stage for initial knowledge graph
107         if (!this.corpus.hasStage(INITIAL_GRAPH_STAGE_NAME, INITIAL_GRAPH_STAGE_VERSION)) {
108             this.initialGraphStage = corpus.makeStage(INITIAL_GRAPH_STAGE_NAME, INITIAL_GRAPH_STAGE_VERSION, noDependencies,
109                     INITIAL_GRAPH_STAGE_DESCRIPTION, false);
110         } else {
111             this.initialGraphStage = this.corpus.getStage(INITIAL_GRAPH_STAGE_NAME, INITIAL_GRAPH_STAGE_VERSION);
112         }
113 
114         // Metadata
115         this.attribMetadata = Concrete.AttributeMetadata.newBuilder().setTool("jhu.hltcoe.rebar2.ingest.TwitterCorpusIngester")
116                 .setConfidence(1.0f).build();
117         this.annotationMetadata = Concrete.AnnotationMetadata.newBuilder().setTool("jhu.hltcoe.rebar2.ingest.TwitterCorpusIngester")
118         // .setTimestamp()
119                 .build();
120     }
121 
122     void close() throws RebarException {
123         LOGGER.info("Closing initializer");
124         this.initializer.close();
125         LOGGER.info("Marking stages as public");
126         corpus.markStagePublic(tweetInfoStage);
127         corpus.markStagePublic(secSegStage);
128         corpus.markStagePublic(sentSegStage);
129         corpus.markStagePublic(initialGraphStage);
130     }

132     public static void main(String[] args) throws RebarException, IOException, FileNotFoundException {
133         if (args.length != 1) {
134             LOGGER.info("Usage: TwitterCorpusMaker <corpusname>");
135             return;
136         }
137 
138         try {
139             String corpusName = args[0];
140             CorpusFactory cf = RebarBackends.ACCUMULO.getCorpusFactory();
141             if (cf.corpusExists(corpusName)) {
142                 LOGGER.warn("Corpus already exists " + corpusName);
143             } else {
144                 LOGGER.info("Creating corpus " + corpusName);
145                 cf.makeCorpus(corpusName);
146                 Corpus corpus = cf.getCorpus(corpusName);
147                 TwitterCorpusMaker stages = new TwitterCorpusMaker(corpus);
148                 LOGGER.info("Closing corpus");
149                 stages.close();
150             }
151         } catch (Exception e) {
152             // We use system.exit here (rather than just returning) to
153             // be *sure* that we exit, even if some accumulo thread
154             // didn't get closed (e.g., because an exception was
155             // raised).
156             LOGGER.error(e, e);
157             System.exit(-1);
158         } finally {
159             System.exit(0);
160         }
161     }
