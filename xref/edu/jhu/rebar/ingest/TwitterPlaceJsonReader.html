35  		if (mapper==null) mapper = new ObjectMapper();
36                  mapper.configure(DeserializationFeature.FAIL_ON_UNKNOWN_PROPERTIES, false);
37  		return mapper.readValue(json, TwitterPlaceJsonReader.class).toProto();
38  	}
