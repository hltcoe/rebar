15  /** Class used to read in a JSON Twitter User record and convert it to
16   * a rebar protobuf object (Concrete.TwitterUser).  The input JSON record
17   * should use the standard twitter API, as defined at:
18   *
19   * https://dev.twitter.com/docs/platform-objects/users
20   */
21  @JsonIgnoreProperties({
22  	"profile_background_tile", "profile_background_color", 
23  	"profile_background_image_url", "profile_background_image_url_https", 
24  	"profile_image_url", "profile_image_url_https", "profile_link_color", 
25  	"profile_text_color", "profile_sidebar_border_color", 
26  	"profile_sidebar_fill_color", "protected",
27  	// The following are deprecated in the Twitter API 
28  	"following", "notifications", "id_str",
29  	// The following are non-standard additions in the mtire data set
30  	"mitre_gender"})
31  class TwitterUserJsonReader {
32  	// Public interface
33  	public static Concrete.TwitterUser parseJson(String json) throws java.io.IOException {
34  		if (mapper==null) mapper = new ObjectMapper();
35  		return mapper.readValue(json, TwitterUserJsonReader.class).toProto();
36  	}
37  
38  	// Private implementation details
39  	private static ObjectMapper mapper = null;
40  	private Concrete.TwitterUser.Builder builder;
41  	TwitterUserJsonReader() {
42  		builder = Concrete.TwitterUser.newBuilder(); }
43  	Concrete.TwitterUser toProto() {
44  		return builder.build(); }
