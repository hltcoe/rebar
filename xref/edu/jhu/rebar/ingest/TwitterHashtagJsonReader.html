17  /** Class used to read in a JSON Twitter Hashtag record and convert it to
18   * a rebar protobuf object (Concrete.TwitterEntities.HashTag).  The input JSON record
19   * should use the standard twitter API, as defined at:
20   *
21   * https://dev.twitter.com/docs/platform-objects/entities
22   */
23  class TwitterHashtagJsonReader {
24  	// Public interface
25  	public static Concrete.TwitterEntities.HashTag parseJson(String json) throws java.io.IOException {
26  		if (mapper==null) mapper = new ObjectMapper();
27                  mapper.configure(DeserializationFeature.FAIL_ON_UNKNOWN_PROPERTIES, false);
28  		return mapper.readValue(json, TwitterHashtagJsonReader.class).toProto();
29  	}
30  
31  	// Private implementation details
32  	private static ObjectMapper mapper = null;
33  	private Concrete.TwitterEntities.HashTag.Builder builder;
34  	TwitterHashtagJsonReader() {
35  		builder = Concrete.TwitterEntities.HashTag.newBuilder(); }
36  	Concrete.TwitterEntities.HashTag toProto() {
37  		return builder.build(); }
38  
39  	// Converters: one property for each expected json field.
40    	@JsonProperty("indices") public void setIndices(List<Integer> indices) {
41            if (indices != null && indices.size() > 0) {
42              builder.setStartOffset(indices.get(0));
43              builder.setEndOffset(indices.get(1));
44            }
45          }
46  	@JsonProperty("text") public void setText(String text) {
47            if (text != null) builder.setText(text); }
