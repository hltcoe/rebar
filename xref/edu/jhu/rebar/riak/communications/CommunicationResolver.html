15   * A class used to resolve conflicts between {@link Communication} objects in
16   * Riak.
17   * 
18   * We currently just merge everything (or attempt to).
19   * 
20   * @author max
21   * 
22   */
23  public class CommunicationResolver implements ConflictResolver<RiakCommunication> {
24

26       * 
27       */
28      public CommunicationResolver() {
29          // TODO Auto-generated constructor stub
30      }
31  
32      @Override
33      public RiakCommunication resolve(Collection<RiakCommunication> siblings) {
34          if (siblings.size() > 1) {
35              // We have siblings, need to resolve them
36              Iterator<RiakCommunication> i = siblings.iterator();
37              RiakCommunication rootRiakComm = i.next();
38              
39              Communication mergedCommObject = rootRiakComm.getComm();
40              while (i.hasNext()) {
41                  mergedCommObject = Communication.newBuilder(mergedCommObject)
42                      .mergeFrom(i.next().getComm())
43                      .build();
44              }
45  
46              return new RiakCommunication(mergedCommObject);
47          } else if (siblings.size() == 1) {
48              // Only one object - just return it
49              return siblings.iterator().next();
50          } else {
51              // No object returned - return null
52              return null;
53          }
54      }
