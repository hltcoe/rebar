28  public class CommunicationReader {
29  
30      private IRiakClient irc;
31      private Bucket corpusBucket;
32      private Bucket commsBucket;
33      
34      /**
35       * @throws RiakException 
36       * 
37       */
38      public CommunicationReader() throws RiakException {
39          this.irc = RiakFactory.pbcClient();
40          
41          this.corpusBucket = this.irc.fetchBucket(RiakConstants.CORPORA_BUCKET_NAME)
42                  .execute();
43          this.commsBucket = this.irc.fetchBucket(RiakConstants.COMMUNICATIONS_BUCKET_NAME)
44                  .execute();
45      }
46      
47      public Set<RiakCommunication> fetchAllComms(String nameOfCorpusToFetch) throws RebarException {
48          Set<RiakCommunication> rcSet = new HashSet<>();
49          
50          RiakCorpus rc = new RiakCorpus(nameOfCorpusToFetch);
51          try {
52              rc = this.corpusBucket.fetch(rc)
53                      .withConverter(new KryoRiakCorpusConverter())
54                      .withResolver(new CorpusResolver())
55                      .execute();
56              for (String commId : rc.getCommIdSet()) {
57                  RiakCommunication rComm = new RiakCommunication(commId);
58                  rComm = this.commsBucket.fetch(rComm)
59                              .withConverter(new CommunicationConverter())
60                              .withResolver(new CommunicationResolver())
61                              .execute();
62                  rcSet.add(rComm);
63              }
64          } catch (UnresolvedConflictException | RiakRetryFailedException | ConversionException e) {
65              throw new RebarException(e);
66          }
67          
68          
69          return rcSet;
70      }
71      
72      public void close() {
73          this.irc.shutdown();
74      }
