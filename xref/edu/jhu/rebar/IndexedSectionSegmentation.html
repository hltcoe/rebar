17  	public static IndexedSectionSegmentation build(Concrete.UUID uuid, ProtoIndex index) throws RebarException
18  	{
19  		IndexedSectionSegmentation cached = index.getIndexedProto(uuid);
20  		if (cached != null) return cached;
21  		else return new IndexedSectionSegmentation((Concrete.SectionSegmentation)index.lookup(uuid), index);
22  	}
23  
24  	public static IndexedSectionSegmentation build(Concrete.SectionSegmentation sectionSeg, ProtoIndex index) throws RebarException {
25  		IndexedSectionSegmentation cached = index.getIndexedProto(sectionSeg.getUuid());
26  		if (cached != null) return cached;
27  		else return new IndexedSectionSegmentation(sectionSeg, index);
28  
29  	}
30  
31  	private IndexedSectionSegmentation(Concrete.SectionSegmentation sectionSeg, ProtoIndex index) throws RebarException
32  	{
33  		super(sectionSeg, index);
34  	}

44  	/** Return an IndexedSection for the section with the given

50  	/** Return a list of IndexedSections for the sections in this
51  	 * section segmentation. */
52  	public List<IndexedSection> getSectionList() throws RebarException {
53  		List<IndexedSection> result = new ArrayList<IndexedSection>();
