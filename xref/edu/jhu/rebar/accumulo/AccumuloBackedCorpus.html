159     /** Given a Communication whose Knowledge Graph may contain
160      * multiple Edge objects with the same edge_id, merge the
161      * attributes of those edges, and remove any duplicates from the
162      * vertex neighbor lists.  This is used when loading a
163      * communication because we might be loading multiple stages that
164      * independently added information for the same edge, in which
165      * case they would each have created an Edge object under the
166      * knowledge graph; but these will not get automatically merged. */
167 	
168     private Concrete.Communication mergeDuplicateEdges(Concrete.Communication com) {
169 	Concrete.KnowledgeGraph graph = com.getKnowledgeGraph();
170 	List<Concrete.Edge> edges = graph.getEdgeList();
171 	if (edges.size() == 0) return com;
172 	Map<Concrete.EdgeId, Concrete.Edge> edgeMap = new HashMap<Concrete.EdgeId, Concrete.Edge>();
173 	Map<Concrete.UUID, Set<Concrete.UUID>> neighbors = new HashMap<Concrete.UUID, Set<Concrete.UUID>>();
174 	boolean mergeIsNeeded = false;
175 	for (Concrete.Edge edge: edges) {
176 	    Concrete.EdgeId edgeId = edge.getEdgeId();
177 	    // Update edgeMap to include this edge.
178 	    if (edgeMap.containsKey(edgeId)) {
179 		Concrete.Edge newEdge = edgeMap.get(edgeId).toBuilder()
180 		    .mergeFrom(edge).build();
181 		edgeMap.put(edgeId, newEdge);
182 		mergeIsNeeded = true;
183 	    } else {
184 		edgeMap.put(edgeId, edge);
185 	    }
186 	    // Update neighbors map
187 	    Concrete.UUID v1 = edge.getEdgeId().getV1();
188 	    Concrete.UUID v2 = edge.getEdgeId().getV2();
189 	    if (!neighbors.containsKey(v1))
190 		neighbors.put(v1, new HashSet<Concrete.UUID>());
191 	    if (!neighbors.containsKey(v2))
192 		neighbors.put(v2, new HashSet<Concrete.UUID>());
193 	    neighbors.get(v1).add(v2);
194 	    neighbors.get(v2).add(v1);
195 	}
196 	/*if (!mergeIsNeeded) return com; */
197 	// Replace the edges/neighbors values.
198 	Concrete.Communication.Builder comBuilder = com.toBuilder();
199 	Concrete.KnowledgeGraph.Builder graphBuilder = comBuilder.getKnowledgeGraphBuilder();
200 	graphBuilder.clearEdge();
201 	graphBuilder.addAllEdge(edgeMap.values());
202 	for (Concrete.Vertex.Builder vbuilder: graphBuilder.getVertexBuilderList()) {
203 	    vbuilder.clearNeighbor();
204 	    vbuilder.addAllNeighbor(neighbors.get(vbuilder.getUuid()));
205 	}
206 	return comBuilder.build();
207     }

210      * communications from protobuf. */
211     private class ComReader 
212 	extends AccumuloProtoReader<Concrete.Communication, Concrete.Communication.Builder, String, IndexedCommunication> 
213     {
214 	ComReader(Collection<Stage> stages) throws RebarException {
215 	    super(accumuloConnector, comsTableName, stages); }
216 	protected Concrete.Communication.Builder getRootBuilder(String comId) {
217 	    return Concrete.Communication.newBuilder(); }
218 	protected IndexedCommunication wrap(String comId, Concrete.Communication comm, Map<Key,Value> row, StageOwnership stageOwnership) 
219 	    throws RebarException
220 	{
221 	    comm = mergeDuplicateEdges(comm);
222 	    return new IndexedCommunication(comm, new ProtoIndex(comm), stageOwnership); }
223 	protected Text toRowId(String identifier) {
224 	    return new Text(identifier); }
225 	protected String toIdentifier(Text rowId) {
226 	    return rowId.toString(); }
227     }

281 	public IndexedCommunication addCommunication(Concrete.Communication communication, boolean flush) 
282 	    throws RebarException 
283 	{
284 	    Text rowId = rowIdFor(communication);
285 	    LOGGER.debug("Row id: " + rowId);
286 	    if (rowExists(rowId))
287 		throw new RebarException("Duplicate communication!");
288 	    // Build the indexed communication (this will be our return value).
289 	    IndexedCommunication com = new IndexedCommunication(communication, new ProtoIndex(communication), null);
290 	    // Make sure that v1<=v2 for all edges in this communication's 
291 	    // graph; and that the neighbor lists are correct.
292 	    IndexedKnowledgeGraph graph = com.getKnowledgeGraph();
293 	    graph.checkEdgeIds();
294 	    graph.checkVertexNeighbors();
295 	    // Write the communication.
296 	    write(rowId, 
297 		  AccumuloBackedCorpus.ROOT_CF,
298 		  AccumuloBackedCorpus.EMPTY_CQ,
299 		  new Value(communication.toByteArray()));
300 	    summaryTable.incrementCount(getName(), "com");
301 	    return com;
302 	}

323 	@Override
324 	    public void saveCommunication(IndexedCommunication com) 
325 	    throws RebarException
326 	{
327 	    Text comId = rowIdFor(com.getProto());
328 	    if (DEBUG_LEVEL>0 && (!rowExists(comId)))
329 		throw new RebarException("Communication "+comId+" not found in corpus");
330 	    // Make sure that v1<=v2 for all edges in this communication's 
331 	    // graph; and that the neighbor lists are correct.
332 	    IndexedKnowledgeGraph graph = com.getKnowledgeGraph();
333 	    graph.checkEdgeIds();
334 	    graph.checkVertexNeighbors();
335 	    // Write the changes.
336 	    writeProtoModifications(stage, comId, com.getIndex());
337 	}

419     public static boolean corpusExists(String corpusName) throws RebarException {
420 	AccumuloConnector accumuloConnector = new AccumuloConnector();
421 	boolean result = true;
422 	for (String suffix: TABLE_SUFFIXES) {
423 	    if (!accumuloConnector.tableExists(TABLE_PREFIX+corpusName+suffix))
424 		result = false;
425 	}
426 	accumuloConnector.release();
427 	return result;
428     }

430     public static Set<String> listCorpora() throws RebarException {
431 	AccumuloConnector accumuloConnector = new AccumuloConnector();
432 	Map<String, Integer> corpusTableCounts = new HashMap<String, Integer>();
433 	for (String tableName: accumuloConnector.listTables()) {
434 	    if (tableName.startsWith(TABLE_PREFIX)) {
435 		for (String suffix: TABLE_SUFFIXES) {
436 		    if (tableName.endsWith(suffix)) {
437 			int s = TABLE_PREFIX.length();
438 			int e = tableName.length()-suffix.length();
439 			String corpusName = tableName.substring(s, e);
440 			Integer oldVal = corpusTableCounts.get(corpusName);
441 			if (oldVal==null) oldVal = 0;
442 			corpusTableCounts.put(corpusName, oldVal+1);
443 		    }
444 		}
445 	    }
446 	}
447 	Set<String> result = new TreeSet<String>();
448 	for (Map.Entry<String,Integer> entry: corpusTableCounts.entrySet()) {
449 	    if (entry.getValue() == TABLE_SUFFIXES.length) {
450 		result.add(entry.getKey());
451 	    } else {
452 		System.err.println("Warning: one or more tables appears to be "+
453 				   "missing for corpus "+entry.getKey());
454 	    }
455 	}
456 	accumuloConnector.release();
457 	return result;
458     }

460     public static void deleteCorpus(String corpusName) throws RebarException {
461 	AccumuloConnector accumuloConnector = new AccumuloConnector();
462 	for (String suffix: TABLE_SUFFIXES)
463 	    accumuloConnector.deleteTable(TABLE_PREFIX+corpusName+suffix);
464 	AccumuloSummaryTable summaryTable = new AccumuloSummaryTable(accumuloConnector, TABLE_PREFIX);
465 	summaryTable.deleteEntry(corpusName);
466 	accumuloConnector.release();
