44  /** An iterator that reads protobuf objects from the Accumulo table.
45   * In particular, for each row, it reads a "root" protobuf object from
46   * a fixed column, and then merges in zero or more additional protobuf
47   * fields (the output of different processing stage). */
48  /*package-private*/ 
49  abstract class AccumuloProtoReader
50  	<ProtoObj extends Message, ProtoBuilder extends Message.Builder, 
51       Identifier, Wrapper extends IndexedProto<ProtoObj>> 
52  {
53  	//----------------------------------------------------------------------
54  	// These are the hooks that subclasses must fill in:
55  	// (constructor)
56  	abstract protected ProtoBuilder getRootBuilder(Identifier rootId);
57  	abstract protected Wrapper wrap(Identifier id, ProtoObj obj, Map<Key,Value> row, StageOwnership stageOwnership) throws RebarException;
58  	abstract protected Text toRowId(Identifier id);
59  	abstract protected Identifier toIdentifier(Text rowId);

107 	// Constructor
108 	AccumuloProtoReader(AccumuloConnector accumuloConnector, String tableName, 
109 						Collection<Stage> inputStages) throws RebarException {
110 		this.tableName = tableName;
111 		this.accumuloConnector = accumuloConnector;
112 		this.inputStages = Collections.unmodifiableSet(new TreeSet<Stage>(inputStages));
113 		this.activeIterators = new HashSet<ProtoIterator>();
114 
115 		// Determine which stages we need to read
116 		stageIds = new TreeMap<Integer, Stage>();
117 		stageIds.put(0, null); // root cell.
118 		for(Stage stage: inputStages)
119 			addStageDependencies(stage, stageIds);
120 	}
121 
122 	/** Add the id of the given stage plus any stages it depends on to `ids`. */
123 	private void addStageDependencies(Stage stage, Map<Integer, Stage> ids) {
124 		if (!ids.containsKey(stage.getStageId())) {
125 			ids.put(stage.getStageId(), stage);
126 			for (Stage dep: stage.getDependencies())
127 				addStageDependencies(dep, ids);
128 		}
129 	}

131 	public Iterator<Wrapper> read(Collection<Identifier> subset, boolean recordStageOwnership) throws RebarException {
132 		ProtoIterator iter = new ProtoIterator(subset, recordStageOwnership);
133 		activeIterators.add(iter);
134 		return iter;
135 	}

137 	public Wrapper read(Identifier id, boolean recordStageOwnership) throws RebarException {
138 		Wrapper result = readIfExists(id, recordStageOwnership);
139 		if (result == null)
140 			throw new RebarException("No element found with id="+id+" found");
141 		//getStageOwnershipFor(id); // TESTING!!!!
142 		return result;
143 	}
144 	
145 	public Wrapper readIfExists(Identifier id, boolean recordStageOwnership) throws RebarException {
146 		Collection<Identifier> subset = new ArrayList<Identifier>(1);
147 		subset.add(id);
148 		ProtoIterator iter = new ProtoIterator(subset, recordStageOwnership);
149 		try {
150 			if (!iter.hasNext())
151 				return null;
152 			return iter.next();
153 		} finally {
154 			iter.close();
155 		}
156 	}

162 	private class ProtoIterator implements Iterator<Wrapper> {
163 		/** The scanner used to read in cells.  Might be a simple scanner
164 		 * or a batch scanner. */
165 		private final ScannerBase scanner;
166 		
167 		/** The scanner's iterator. */
168 		private final Iterator<Map.Entry<Key, Value>> iterator;
169 
170 		/** Has this iterator been closed? */
171 		boolean closed;
172 
173 		/** Should we record the stage ownership for each value? */
174 		boolean recordStageOwnership;
175 		
176 		public ProtoIterator(Collection<Identifier> subset, boolean recordStageOwnership) throws RebarException {
177 			this.scanner = createScanner(subset);
178 			this.iterator = scanner.iterator();
179 			this.recordStageOwnership = recordStageOwnership;
180 			this.closed = false;
181 		}
182 
183 		@Override
184 		public boolean hasNext() { 
185 			if (iterator.hasNext()) {
186 				return true;
187 			} else {
188 				if (!closed) close(); // auto-close at end of iteration.
189 				return false;
190 			}
191 		}
192 		@Override
193 		public void remove() { 
194 			throw new UnsupportedOperationException("You cannot remove with this iterator."); 
195 		}
196 		@Override
197 		public Wrapper next() {
198 			final Map.Entry<Key,Value> row = iterator.next();
199 			try {
200 				return parseRow(row.getKey(), row.getValue(), recordStageOwnership);
201 			} catch (RebarException e) {
202 				throw new RuntimeException(e);
203 			}
204 		}
205 
206 		public void close() {
207 			if (!closed) { // This is safe to call multiple times.
208 				if (scanner instanceof BatchScanner) {
209 					((BatchScanner)scanner).close();
210 				}
211 				activeIterators.remove(this);
212 				closed = true;
213 			}
214 		}
215 
216 		private ScannerBase createScanner(Collection<Identifier> subset) throws RebarException {
217 			ScannerBase scanner;
218 			if (subset instanceof AccumuloAllRowIdsCollection) {
219 				scanner = accumuloConnector.createScanner(tableName);
220 			} else { 
221 				BatchScanner bs = accumuloConnector.createBatchScanner(tableName);
222 				bs.setRanges(rangesForSubset(subset));
223 				scanner = bs;
224 			}
225 
226 			// Fetch the columns for each stage we are interested in.
227 			// Each stage's output will be in a column whose column
228 			// family is a fixed prefix plus the stage id number; and
229 			// whose column qualifier is a UUID.
230 			for (int stageId: stageIds.keySet()) {
231 				byte[] cf = ByteUtil.fromIntWithPrefix(STAGE_CF_PREFIX, stageId);
232 				scanner.fetchColumnFamily(new Text(cf));
233 			}
234 			configureScanner(scanner);
235 			// Use WholeRowIterator to fetch a single row at a time.
236 			scanner.addScanIterator(new IteratorSetting(WHOLE_ROW_ITERATOR_PRIORITY, 
237 														"wholeRows", WholeRowIterator.class));
238 			return scanner;
239 		}
240 
241 		private Collection<Range> rangesForSubset(Collection<Identifier> identifiers) {
242 			Collection<Range> ranges = new ArrayList<Range>(identifiers.size());
243 			for(Identifier identifier: identifiers)
244 				ranges.add(new Range(toRowId(identifier)));
245 			return ranges;
246 		}
247 	}

261 	private class StageOutput {
262 		public final Stage stage;
263 		public final byte[] protobufBytes;
264 		public StageOutput(Stage stage, byte[] protobufBytes) {
265 			this.stage = stage; this.protobufBytes = protobufBytes; }
266 	}

268 	protected ProtoObj getRoot(Identifier rootId, Map<Key,Value> decodedRow) 
269 		throws RebarException
270 	{
271 		ProtoBuilder rootBuilder = getRootBuilder(rootId);
272 		// In some special cases (notably edges) there will not be any
273 		// column with cf=ROOT_CF.
274 		for (Map.Entry<Key,Value> entry: decodedRow.entrySet()) {
275 			if (entry.getKey().compareColumnFamily(ROOT_CF)==0) {
276 				assert(entry.getKey().compareColumnQualifier(EMPTY_CQ)==0);
277 				// Merge in the root object.
278 				try {
279 					rootBuilder.mergeFrom(entry.getValue().get());
280 				} catch (InvalidProtocolBufferException e) {
281 					throw new RebarException(e);
282 				}
283 			}
284 		}
285 		// rootBuilder.build() is required to return a ProtoObj.
286 		@SuppressWarnings("unchecked")
287 		ProtoObj root = (ProtoObj)(rootBuilder.build());
288 		return root;
289 	}

292 	private Wrapper parseRow(Key k, Value v, boolean recordStageOwnership) throws RebarException {
293 		//System.err.print("ROW ");
294 		//for (byte b: dr.keySet().iterator().next().getRow().getBytes()) 
295 		//	System.err.print(" "+b);
296 		//System.err.println();
297 
298 		Identifier rootId = toIdentifier(k.getRow());
299 
300 
301 		Map<Key,Value> decodedRow = decodeRow(k, v);
302 
303 		ProtoObj root = getRoot(rootId, decodedRow);
304 
305 		// Allocate stage ownership map (if we're generating it)
306 		StageOwnership stageOwnership = recordStageOwnership?new StageOwnership():null;
307 
308 		// If that's all we have, then return it.
309 		Iterator<Map.Entry<Key,Value>> cellIter = decodedRow.entrySet().iterator();
310 		Map.Entry<Key,Value> cell = getNextStageCell(cellIter);
311 		if (cell == null) {
312 			if (stageOwnership != null) stageOwnership.freeze();
313 			return wrap(rootId, root, decodedRow, stageOwnership);
314 		}
315 
316 		// Index the remaining messages by uuid.
317 		Map<ProtoIndex.ModificationTarget, List<StageOutput>> stageValues = 
318 		        new HashMap<ProtoIndex.ModificationTarget, List<StageOutput>>();
319 		for (; cell != null; cell=getNextStageCell(cellIter)) {
320 			ProtoIndex.ModificationTarget target = 
321 				new ProtoIndex.ModificationTarget(cell.getKey().getColumnQualifier().getBytes());
322 			int stageId = ByteUtil.toIntWithPrefix(STAGE_CF_PREFIX, cell.getKey().getColumnFamily().getBytes());
323 			Stage stage = stageIds.get(stageId);
324 			List<StageOutput> values = stageValues.get(target);
325 			if (values == null) {
326 				values = new ArrayList<StageOutput>(1);
327 				stageValues.put(target, values);
328 			} 
329 			values.add(new StageOutput(stage, cell.getValue().get()));
330 		}
331 
332 		// Merge in stage outputs.
333 		Message mergedRoot = mergeStages(root, stageValues, stageOwnership, null);
334 		if (!stageValues.isEmpty()) {
335 			throw new RebarException(stageValues.size()+" stage output(s) not merged: "
336 									 +stageValues.keySet());
337 		}
338 
339 		@SuppressWarnings("unchecked")
340 		ProtoObj result = ((mergedRoot==null) ? root : (ProtoObj)(mergedRoot));
341 		if (stageOwnership != null) stageOwnership.freeze();
342 		return wrap(rootId, result, decodedRow, stageOwnership);
343 	}

345 	/** Return the next cell that contains the output of a stage.
346 	 * These are identified as cells whose column family begins with
347 	 * STAGE_CF_PREFIX. */
348 	private Map.Entry<Key,Value> getNextStageCell(Iterator<Map.Entry<Key,Value>> cellIter) {
349 		while (cellIter.hasNext()) {
350 			Map.Entry<Key,Value> cell = cellIter.next();
351 			Text cf = cell.getKey().getColumnFamily();
352 			if ((cf.charAt(0)==STAGE_CF_PREFIX) && (!cf.equals(ROOT_CF)))
353 				return cell;
354 		}
355 		return null;
356 	}

358 	/** Merge the given "stageOutput" into the builder "target", and
359 	 * update the list "stageOwnership" to record the stage ownership
360 	 * of each added field.
361 	 *
362 	 * Note: 'target' and 'stageOwnership' are "output" parameters,
363 	 * and are mutated by this method. */
364 	private void mergeAndRecordStageOwnership(Message newFields, Stage outputStage, 
365 											  StageOwnership.FieldValuePointer fvp,
366 											  Message.Builder target, StageOwnership stageOwnership)
367 		throws RebarException, InvalidProtocolBufferException
368 	{
369 		// Add the fields one at a time, adding a StageOwnershipRecord for each value.
370 		for(Map.Entry<FieldDescriptor,Object> field: newFields.getAllFields().entrySet()) {
371 			FieldDescriptor fd = field.getKey();
372 			if (fd.isRepeated()) {
373 				// Append field value.
374 				int origLen = target.getRepeatedFieldCount(fd);
375 				@SuppressWarnings("unchecked")
376 				List<Message> children = (List<Message>) field.getValue();
377 				for (int i=0; i<children.size(); i++) {
378 					target.addRepeatedField(fd, children.get(i));
379 					stageOwnership.put(new StageOwnership.FieldValuePointer(fvp, fd, i+origLen),
380 									   outputStage);
381 				}
382 			} else if (target.hasField(fd)) {
383 				if (fd.getType() == FieldDescriptor.Type.MESSAGE) {
384 					Message newChild = (Message)(field.getValue());
385 					if (IdUtil.getUUIDOrNull(newChild) != null)
386 						throw new RebarException("Stage "+outputStage+" set a non-repeated "+
387 												 "field with a UUID that already has a value: "+
388 												 fd.getFullName());
389 					// Merge field value.
390 					Message.Builder childTarget = ((Message)(target.getField(fd))).toBuilder();
391 					mergeAndRecordStageOwnership(newChild, outputStage,
392 												 new StageOwnership.FieldValuePointer(fvp, fd, 0),
393 												 childTarget, stageOwnership);
394 					target.setField(fd, childTarget.build());
395 				} else {
396 					throw new RebarException("Stage "+outputStage+" set a non-repeated field "+
397 											 "that already has a value: "+fd.getFullName());
398 				}
399 			} else {
400 				target.setField(fd, field.getValue());
401 				stageOwnership.put(new StageOwnership.FieldValuePointer(fvp, fd, 0), outputStage);
402 			}
403 		}
404 	}

408 	/** Returns null if nothing changed, or the new value if we
409 	 * merged something in. */
410 	private Message mergeStages(Message msg, Map<ProtoIndex.ModificationTarget, List<StageOutput>> stageValues,
411 								StageOwnership stageOwnership, StageOwnership.FieldValuePointer fvp)
412 		throws RebarException
413 	{
414 		Message.Builder builder = null;
415 
416 		// Get the modification target for this message.
417 		final ProtoIndex.ModificationTarget target;
418 		Concrete.UUID uuid = IdUtil.getUUIDOrNull(msg); // uuid may be null.
419 		if (uuid != null)
420 			target = new ProtoIndex.ModificationTarget(uuid);
421 		else if (msg instanceof Concrete.Edge) {
422 			target = new ProtoIndex.ModificationTarget(((Concrete.Edge)msg).getEdgeId());
423 		}
424 		else
425 			target = null;
426 		//System.err.println("Merging, target="+target);
427 		// Get any values we should merge into this target, and merge them.
428 		List<StageOutput> valuesToMerge = stageValues.remove(target);
429 		if (valuesToMerge != null) {
430 			try {
431 				// Merge in the new values.
432 				for (StageOutput stageOutput: valuesToMerge) {
433 					if (builder==null) builder = msg.toBuilder();
434 					if (stageOwnership == null) {
435 						builder.mergeFrom(stageOutput.protobufBytes);
436 					} else {
437 						Message newFields = msg.newBuilderForType()
438 							.mergeFrom(stageOutput.protobufBytes).buildPartial();
439 						mergeAndRecordStageOwnership(newFields, stageOutput.stage, 
440 													 fvp, builder, stageOwnership);
441 					}
442 				}
443 			} catch (InvalidProtocolBufferException e) {
444 				throw new RebarException(e);
445 			}
446 			// If we're done with everything, then return immediately.
447 			if (stageValues.isEmpty()) return builder.build();
448 			// If the stage we just merged in is depended on by other
449 			// stages, then we need to rebuild the message before we
450 			// traverse it...  Currently we don't keep that info, so
451 			// for now, just always do it.
452 			if (true) {
453 				msg = builder.build();
454 				builder = null;
455 			}
456 		}
457 
458 		// Recurse to subfields
459 		for(Map.Entry<FieldDescriptor,Object> field: msg.getAllFields().entrySet()) {
460 			FieldDescriptor fd = field.getKey();
461 			if (fd.getJavaType() == FieldDescriptor.JavaType.MESSAGE) {
462 				if (fd.isRepeated()) {
463 					@SuppressWarnings("unchecked")
464 					List<Message> children = (List<Message>) field.getValue();
465 					for (int i=0; i<children.size(); i++) {
466 						Message child = children.get(i);
467 						StageOwnership.FieldValuePointer childFvp = 
468 							(stageOwnership==null)?null:new StageOwnership.FieldValuePointer(fvp, fd, i);
469 						Message mergedChild = mergeStages(child, stageValues, stageOwnership, childFvp);
470 						if (mergedChild != null) {
471 							if (builder==null) builder = msg.toBuilder();
472 							builder.setRepeatedField(fd, i, mergedChild);
473 							if (stageValues.isEmpty()) return builder.build();
474 						}
475 					}
476 				} else {
477 					Message child = (Message) field.getValue();
478 					StageOwnership.FieldValuePointer childFvp = 
479 						(stageOwnership==null)?null:new StageOwnership.FieldValuePointer(fvp, fd, 0);
480 					Message mergedChild = mergeStages(child, stageValues, stageOwnership, childFvp);
481 					if (mergedChild != null) {
482 						if (builder==null) builder = msg.toBuilder();
483 						builder.setField(fd, mergedChild);
484 						if (stageValues.isEmpty()) return builder.build();
485 					}
486 				}
487 			}
488 		}
489 
490 		if (builder != null)
491 			return builder.build();
492 		else if (valuesToMerge != null)
493 			return msg;
494 		else
495 			return null;
496 	}
