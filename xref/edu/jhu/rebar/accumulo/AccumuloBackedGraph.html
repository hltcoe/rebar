42  /**
43   * Accumulo-backed implementation of the Graph interface.  This class
44   * should not be directly instantiated; use Graph.Factory.getGraph()
45   * instead. 
46   * 
47   * The accumulo table that backs each Graph can contain the following
48   * cells:
49   *
50   * <pre>
51   *   RowId    ColFam        ColQual          Value
52   * --------- ------------- -------------- -----------------------
53   *  edgeId    S+stage       target_uuid       protobuf
54   *  edgeId    N+stage       nbr_uuid
55   * </pre>
56   */
57  // key.cf = stageid; key.cq = uuid
58  public class AccumuloBackedGraph extends AccumuloBackedStagedDataCollection implements Graph {

156 	/** Private subclass of AccumuloProtoReader that is used to read
157 	 * vertices from protobuf. */
158 	private class VertexReader 
159 		extends AccumuloProtoReader<Concrete.Vertex, Concrete.Vertex.Builder, Concrete.UUID, IndexedVertex> 
160 	{
161 		private final Graph.Reader graphReader;
162 		VertexReader(Collection<Stage> stages, Graph.Reader graphReader) throws RebarException {
163 			super(accumuloConnector, verticesTableName, stages); 
164 			this.graphReader = graphReader;
165 		}
166 		protected Concrete.Vertex.Builder getRootBuilder(Concrete.UUID vertexId) {
167 			return Concrete.Vertex.newBuilder(); 
168 		}
169 		protected void configureScanner(ScannerBase scanner) {
170             for (int stageId: stageIds.keySet()) {
171                 byte[] cf = ByteUtil.fromIntWithPrefix(NEIGHBOR_CF_PREFIX, stageId);
172                 scanner.fetchColumnFamily(new Text(cf));
173             }
174 		}
175 		protected Text toRowId(Concrete.UUID id) {
176 			return vertexIdToRowId(id); 
177 		}
178 		protected Concrete.UUID toIdentifier(Text rowId) { 
179 			return rowIdToVertexId(rowId); 
180 		}
181 		protected IndexedVertex wrap(Concrete.UUID uuid, Concrete.Vertex vertex, Map<Key,Value> row, StageOwnership stageOwnership) 
182 			throws RebarException
183 		{
184 			// Add in the list of neighbors.
185 			Concrete.Vertex.Builder vertexBuilder = vertex.toBuilder();
186 			for (Map.Entry<Key,Value> cell: row.entrySet()) {
187 				Key key = cell.getKey();
188 				if (cell.getKey().getColumnFamily().charAt(0)==NEIGHBOR_CF_PREFIX) {
189 					Concrete.UUID nbr = ByteUtil.toUUID(cell.getKey().getColumnQualifier().getBytes());
190 					vertexBuilder.addNeighbor(nbr);
191 				}
192 			}
193 			Concrete.Vertex vertexWithNeighbors = vertexBuilder.build();
194 			ProtoIndex protoIndex = new ProtoIndex(vertexWithNeighbors);
195 			return new IndexedVertex(vertexWithNeighbors, protoIndex, stageOwnership, graphReader); 
196 		}
197 	}

224 	/** Reader implementation */
225 	private class Reader implements Graph.Reader {
226 		private VertexReader vertexReader;
227 		private EdgeReader edgeReader;
228 		private final boolean loadStageOwnership;
229 		Reader(Collection<Stage> stages, boolean loadStageOwnership) throws RebarException {
230 			vertexReader = new VertexReader(stages, this); 
231 			edgeReader = new EdgeReader(stages, this);
232 			this.loadStageOwnership = loadStageOwnership;
233 		}
234 		// Vertex reading methods.
235 		@Override
236 		public IndexedVertex loadVertex(Concrete.UUID vertexId) throws RebarException {
237 			return vertexReader.read(vertexId, loadStageOwnership); }
238 		@Override
239 		public Iterator<IndexedVertex> loadVertices(Collection<Concrete.UUID> vertexIds) throws RebarException {
240 			return vertexReader.read(vertexIds, loadStageOwnership); }
241 		@Override
242 		public Iterator<IndexedVertex> loadVertices() throws RebarException {
243 			return vertexReader.read(lookupVertexIdSet("all"), loadStageOwnership); }
244 
245 		// Edge reading methods.
246 		@Override
247 		public IndexedEdge loadEdge(Concrete.EdgeId edgeId) throws RebarException {
248 			IndexedEdge result = edgeReader.readIfExists(edgeId, loadStageOwnership); 
249 			if (result == null) {
250 				Concrete.Edge edge = getCoreEdgeBuilder(edgeId).build();
251 				StageOwnership stageOwnership = loadStageOwnership ? new StageOwnership() : null;
252 				result = new IndexedEdge(edge, new ProtoIndex(edge), 
253 										 stageOwnership, this);
254 			}
255 			return result;
256 		}
257 		@Override
258 		public IndexedEdge loadEdge(Concrete.DirectedEdgeId directedEdgeId) throws RebarException {
259 			Concrete.EdgeId edgeId = IdUtil.buildEdgeId(directedEdgeId);
260 			IndexedEdge edge = loadEdge(edgeId);
261 			if (directedEdgeId.getDst().equals(edgeId.getV1()))
262 				edge = edge.reversed();
263 			return edge;
264 		}
265 		@Override
266 		public IndexedEdge loadEdge(Concrete.UUID src, Concrete.UUID dst) throws RebarException {
267 			return loadEdge(IdUtil.buildDirectedEdgeId(src, dst)); }
268 		@Override
269 		public IndexedEdge loadEdge(IndexedVertex src, IndexedVertex dst) throws RebarException {
270 			return loadEdge(src.getUuid(), dst.getUuid()); }
271 		@Override
272 		public Iterator<IndexedEdge> loadEdges(Collection<Concrete.EdgeId> edgeIds) throws RebarException {
273 			return edgeReader.read(edgeIds, loadStageOwnership); }
274 		@Override
275 		public Iterator<IndexedEdge> loadEdges() throws RebarException {
276 			return edgeReader.read(lookupEdgeIdSet("all"), loadStageOwnership); }
277 
278 		@Override
279 		public void close() throws RebarException {
280 			vertexReader.close(); 
281 			edgeReader.close();
282 		}
283 		@Override
284 		public Collection<Stage> getInputStages() throws RebarException {
285 			return vertexReader.getInputStages(); }
286 		@Override
287 		public Graph getGraph() { return AccumuloBackedGraph.this; }
288 	}

294 	private class Initializer 
295 		extends AccumuloWriter 
296 		implements Graph.Initializer 
297 	{
298 		Initializer() throws RebarException { 
299 			super(accumuloConnector, verticesTableName); 
300 		}
301 
302 		public IndexedVertex addVertex(Concrete.Vertex vertex) throws RebarException {
303 			if (vertex.getNeighborCount() > 0)
304 				throw new RebarException("Graph.Initailizer may not be used to add a vertex with"+
305 										 " neighbors; Edges must be added with a Graph.Writer.");
306 			Text vertexId = rowIdFor(vertex);
307 			if (DEBUG_LEVEL>0 && (rowExists(vertexId)))
308 				throw new RebarException("Vertex "+vertexId+" has already been added!");
309 			write(vertexId,
310 				  AccumuloBackedGraph.ROOT_CF,
311 				  AccumuloBackedGraph.EMPTY_CQ,
312 				  new Value(vertex.toByteArray()));
313 			summaryTable.incrementCount(getName(), "vertex");
314 			return new IndexedVertex(vertex, new ProtoIndex(vertex), null, null);
315 		}
316 	}

322 	private class Writer implements Graph.Writer 
323 	{ 
324 		private final AccumuloWriter vertexWriter;
325 		private final AccumuloWriter edgeWriter;
326 		private final Stage stage;
327 
328 		public Writer(Stage stage) throws RebarException { 
329 			this.vertexWriter = new AccumuloWriter(accumuloConnector, verticesTableName);
330 			this.edgeWriter = new AccumuloWriter(accumuloConnector, edgesTableName);
331 			this.stage = stage;
332 		}
333 
334 		public void close() throws RebarException {
335 			vertexWriter.close();
336 			edgeWriter.close();
337 		}
338 
339 		public void flush() throws RebarException {
340 			vertexWriter.flush();
341 			edgeWriter.flush();
342 		}
343 
344 		public Stage getOutputStage() { return stage; }
345 
346 		public void saveVertex(IndexedVertex vertex) throws RebarException
347 		{
348 			Text rowId = rowIdFor(vertex.getProto());
349 			if (DEBUG_LEVEL>0 && (!vertexWriter.rowExists(rowId)))
350 				throw new RebarException("Vertex "+rowId+" not found in graph");
351 			vertexWriter.writeProtoModifications(stage, rowId, vertex.getIndex());
352 		}
353 
354 		public void saveEdge(IndexedEdge edge) throws RebarException
355 		{
356 			Text rowId = rowIdFor(edge.getProto());
357 			edgeWriter.writeProtoModifications(stage, rowId, edge.getIndex());
358 			// Record the neighborship relationships implied by this edge.
359 			Text v1 = new Text(ByteUtil.fromUUID(edge.getEdgeId().getV1()));
360 			Text v2 = new Text(ByteUtil.fromUUID(edge.getEdgeId().getV2()));
361 			Text cf = new Text(ByteUtil.fromIntWithPrefix(NEIGHBOR_CF_PREFIX, stage.getStageId()));
362 			vertexWriter.write(v1, cf, v2, EMPTY_VALUE);
363 			vertexWriter.write(v2, cf, v1, EMPTY_VALUE);
364 		}
365 	}

390 			edgeWriter.flush();
391 		}

393 		public void saveVertexDiff(Concrete.UUID vertexId, Map<ProtoIndex.ModificationTarget, byte[]> changes) throws RebarException {
394 			Text rowId = vertexIdToRowId(vertexId);
395 			if (DEBUG_LEVEL>0 && (!vertexWriter.rowExists(rowId)))
396 				throw new RebarException("Vertex "+vertexId+" not found in graph");
397 			vertexWriter.writeProtoModifications(stage, rowId, changes);
398 		}
399 
400 		public void saveEdgeDiff(Concrete.EdgeId edgeId, Map<ProtoIndex.ModificationTarget, byte[]> changes) throws RebarException {
401 			Text rowId = edgeIdToRowId(edgeId);
402 			edgeWriter.writeProtoModifications(stage, rowId, changes);
403 			// Record the neighborship relationships implied by this edge.
404 			Text v1 = new Text(ByteUtil.fromUUID(edgeId.getV1()));
405 			Text v2 = new Text(ByteUtil.fromUUID(edgeId.getV2()));
406 			Text cf = new Text(ByteUtil.fromIntWithPrefix(NEIGHBOR_CF_PREFIX, stage.getStageId()));
407 			vertexWriter.write(v1, cf, v2, EMPTY_VALUE);
408 			vertexWriter.write(v2, cf, v1, EMPTY_VALUE);
409 		}
410 	}

512 	private static Text vertexIdToRowId(Concrete.UUID id) {
513 		return new Text(ByteUtil.fromUUID(id));
514 	}
515 
516 	private static Concrete.UUID rowIdToVertexId(Text rowId) {
517 		return ByteUtil.toUUID(rowId.getBytes());
518 	}
519 
520 	private static Text edgeIdToRowId(Concrete.EdgeId id) {
521 		if ((id.getV1() == ZERO_UUID) || (id.getV2() == ZERO_UUID))
522 			throw new IllegalArgumentException("Attempt to write a edge with an empty uuid");
523 		if (!IdUtil.edgeIdIsValid(id))
524 			throw new IllegalArgumentException("Attempt to write edge where v2<v1: "+id);
525 		byte[] rowid = new byte[32];
526 		ByteUtil.fromLong(id.getV1().getHigh(),   0, rowid);
527 		ByteUtil.fromLong(id.getV1().getLow(),    8, rowid);
528 		ByteUtil.fromLong(id.getV2().getHigh(),  16, rowid);
529 		ByteUtil.fromLong(id.getV2().getLow(),   24, rowid);
530 		return new Text(rowid);
531 	}
532 
533 	private static Concrete.EdgeId rowIdToEdgeId(Text rowId) {
534 		// Will always return a normalized EdgeId.
535 		byte[] bytes = rowId.getBytes();
536 		Concrete.UUID v1 = Concrete.UUID.newBuilder()
537 			.setHigh(ByteUtil.toLong(bytes, 0))
538 			.setLow(ByteUtil.toLong(bytes, 8))
539 			.build();
540 		Concrete.UUID v2 = Concrete.UUID.newBuilder()
541 			.setHigh(ByteUtil.toLong(bytes, 16))
542 			.setLow(ByteUtil.toLong(bytes, 24))
543 			.build();
544 		return Concrete.EdgeId.newBuilder()
545 			.setV1(v1).setV2(v2).build();
546 	}
547 
548 	private static Text rowIdFor(Concrete.Vertex vertex) throws RebarException {
549 		Concrete.UUID uuid = IdUtil.getUUID(vertex);
550 		if (uuid == ZERO_UUID)
551 			throw new RebarException("Attempt to write a vertex with an empty uuid");
552 		return vertexIdToRowId(uuid);
553 	}
554 
555 	private static Text rowIdFor(Concrete.Edge edge) throws RebarException {
556 		// Note: we do a sanity check that v1<v2 in edgeIdToRowId.
557 		return edgeIdToRowId(edge.getEdgeId());
558 	}

564 	public static boolean graphExists(String graphName) throws RebarException {
565 		AccumuloConnector accumuloConnector = new AccumuloConnector();
566 		boolean result = true;
567 		for (String suffix: TABLE_SUFFIXES) {
568 			if (!accumuloConnector.tableExists(TABLE_PREFIX+graphName+suffix))
569 				result = false;
570 		}
571 		accumuloConnector.release();
572 		return result;
573 	}

575 	public static Collection<String> listGraphs() throws RebarException {
576 		AccumuloConnector accumuloConnector = new AccumuloConnector();
577 		Map<String, Integer> graphTableCounts = new HashMap<String, Integer>();
578 		for (String tableName: accumuloConnector.listTables()) {
579 			if (tableName.startsWith(TABLE_PREFIX)) {
580 				for (String suffix: TABLE_SUFFIXES) {
581 					if (tableName.endsWith(suffix)) {
582 						int s = TABLE_PREFIX.length();
583 						int e = tableName.length()-suffix.length();
584 						String graphName = tableName.substring(s, e);
585 						Integer oldVal = graphTableCounts.get(graphName);
586 						if (oldVal==null) oldVal = 0;
587 						graphTableCounts.put(graphName, oldVal+1);
588 					}
589 				}
590 			}
591 		}
592 		Collection<String> result = new TreeSet<String>();
593 		for (Map.Entry<String,Integer> entry: graphTableCounts.entrySet()) {
594 			if (entry.getValue() == TABLE_SUFFIXES.length) {
595 				result.add(entry.getKey());
596 			} else {
597 				System.err.println("Warning: one or more tables appears to be "+
598 								   "missing for graph "+entry.getKey());
599 			}
600 		}
601 		accumuloConnector.release();
602 		return result;
603 	}
