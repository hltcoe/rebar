29  /** An interface to an "id set table", which associates symbolic names
30   * with sets of item identifers. */
31  /*package-private*/ 
32  abstract class AccumuloIdSetTable<Identifier> {
33  	// ======================================================================
34  	// Hooks for Subclass to fill in
35  	// ======================================================================
36  	
37  	abstract protected Text idToRowId(Identifier id);
38  	abstract protected Identifier rowIdToId(Text id);
39  
40  	// ======================================================================
41  	// Constants
42  	// ======================================================================
43  
44  	private final static Text ID_CF = new Text("id");
45  	private final static Text CTIME_CF = new Text("ctime");
46  	private final static Text EMPTY_CQ = new Text();
47  	private final static Value EMPTY_VALUE = new Value(new byte[]{});
48  
49  	// ======================================================================
50  	// Private Variables
51  	// ======================================================================
52  
53  	/** The name of the accumulo table that backs this AccumuloIdSetTable */
54  	private final String tableName;
55  	/** The connection to accumulo */
56  	private final AccumuloConnector accumuloConnector;
57  
58  	// ======================================================================
59  	// Constructor
60  	// ======================================================================
61  
62  	AccumuloIdSetTable(AccumuloConnector accumuloConnector, String tableName) {
63  		this.accumuloConnector = accumuloConnector;
64  		this.tableName = tableName; // = corpusName+IDSETS_TABLE_SUFFIX
65  	}
66  
67  	// ======================================================================
68  	// Public Methods
69  	// ======================================================================
70  
71  	public void registerIdSet(String name, Collection<Identifier> idSet) throws RebarException {
72  		BatchWriter writer = accumuloConnector.createBatchWriter(tableName);
73  		Mutation m = new Mutation(new Text(name));
74  		for (Identifier id: idSet)
75  			m.put(ID_CF, idToRowId(id), EMPTY_VALUE);
76  		m.put(CTIME_CF, EMPTY_CQ, new Value(new Date().toString().getBytes()));
77  		try {
78  			writer.addMutation(m);
79  			writer.flush();
80  			writer.close();
81  		} catch (MutationsRejectedException e) {
82  			throw new RebarException(e);
83  		}
84  	}
85  
86  	public Collection<Identifier> lookupIdSet(String name) throws RebarException {
87  		HashSet<Identifier> result = new HashSet<Identifier>();
88  		final Scanner scanner = accumuloConnector.createScanner(tableName);
89  		scanner.fetchColumnFamily(ID_CF);
90  		scanner.setRange(new Range(new Text(name)));
91  		final Iterator<Map.Entry<Key, Value>> iterator = scanner.iterator();
92  		while (iterator.hasNext())
93  			result.add(rowIdToId(iterator.next().getKey().getColumnQualifier()));
94  		return result;
95  	}
96  		
97  	public Collection<String> getSubsetNames() throws RebarException {
98  		HashSet<String> names = new HashSet<String>();
99  		final Scanner scanner = accumuloConnector.createScanner(tableName);
100 		scanner.fetchColumnFamily(CTIME_CF);
101 		scanner.addScanIterator(new IteratorSetting(1001, "key_only",
102 													SortedKeyIterator.class));
103 		final Iterator<Map.Entry<Key, Value>> iterator = scanner.iterator();
104 		while (iterator.hasNext()) {
105 			names.add(iterator.next().getKey().getRow().toString());
106 		}
107 		return names;
108 	}
