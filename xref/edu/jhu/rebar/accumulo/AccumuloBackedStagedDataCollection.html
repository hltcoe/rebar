14  import edu.jhu.hlt.concrete.util.ByteUtil;
15  import edu.jhu.rebar.RebarException;

19  /**
20   * Base class for the Accumulo backed Corpus and Graph classes. This class is
21   * not indended to be directly instantiated.
22   */
23  public class AccumuloBackedStagedDataCollection implements StagedDataCollection {
24      // ======================================================================
25      // Constants
26      // ======================================================================
27      // We use a one-byte prefix for each column family type.
28      protected final static byte STAGE_CF_PREFIX = 10;
29      protected final static String STAGES_TABLE_SUFFIX = "_stages";
30  
31      // ======================================================================
32      // Private Variables
33      // ======================================================================
34      protected final AccumuloStageTable stageTable;
35      protected final AccumuloConnector accumuloConnector;
36      protected final AccumuloSummaryTable summaryTable;
37      protected final String name;
38      protected final String[] tableNames;
39      protected final String[] protoTableNames;
40      protected final byte[] stageCFPrefixes;
41  
42      // ======================================================================
43      // Constructor
44      // ======================================================================
45  
46      protected AccumuloBackedStagedDataCollection(String name, boolean create, String tablePrefix, String[] tableSuffixes,
47              String[] protoTableSuffixes, byte[] stageCFPrefixes) throws RebarException {
48          this.name = name;
49          this.accumuloConnector = new AccumuloConnector();
50          if (create) {
51              for (String suffix : tableSuffixes)
52                  accumuloConnector.createTable(tablePrefix + name + suffix);
53          } else {
54              for (String suffix : tableSuffixes) {
55                  if (!accumuloConnector.tableExists(tablePrefix + name + suffix)) {
56                      throw new RebarException("Table " + tablePrefix + name + suffix + " not found; is the name \"" + name + "\" correct?");
57                  }
58              }
59          }
60          this.stageTable = new AccumuloStageTable(accumuloConnector, tablePrefix + name + STAGES_TABLE_SUFFIX, this);
61          this.summaryTable = new AccumuloSummaryTable(accumuloConnector, tablePrefix);
62          this.tableNames = new String[tableSuffixes.length];
63          for (int i = 0; i < tableSuffixes.length; i++)
64              tableNames[i] = tablePrefix + name + tableSuffixes[i];
65          this.protoTableNames = new String[protoTableSuffixes.length];
66          for (int i = 0; i < protoTableSuffixes.length; i++)
67              protoTableNames[i] = tablePrefix + name + protoTableSuffixes[i];
68          this.stageCFPrefixes = stageCFPrefixes;
69      }
70  
71      // ======================================================================
72      // Interface Methods
73      // ======================================================================
74  
75      @Override
76      public void close() throws RebarException {
77          summaryTable.close();
78          accumuloConnector.release();
79      }
80  
81      @Override
82      public String getName() {
83          return name;
84      }
85  
86      @Override
87      public Stage getStage(String stageName, String stageVersion) throws RebarException {
88          return stageTable.getStage(stageName, stageVersion);
89      }
90  
91      @Override
92      public Stage getStage(int stageId) throws RebarException {
93          return stageTable.getStage(stageId);
94      }
95  
96      @Override
97      public Stage makeStage(String stageName, String stageVersion, Set<Stage> dependencies, String description, boolean deleteIfExists)
98              throws RebarException {
99          return stageTable.addStage(stageName, stageVersion, dependencies, description, deleteIfExists);
100     }
101 
102     @Override
103     public SortedSet<Stage> getStages() throws RebarException {
104         return stageTable.getStages();
105     }
106 
107     @Override
108     public Stage getStage(String stageString) throws RebarException {
109         return stageTable.getStage(stageString);
110     }
111 
112     @Override
113     public boolean hasStage(String stageName, String stageVersion) throws RebarException {
114         return stageTable.hasStage(stageName, stageVersion);
115     }
116 
117     @Override
118     public void markStagePublic(Stage stage) throws RebarException {
119         stageTable.markStagePublic(stage);
120     }
121 
122     @Override
123     public void deleteStage(Stage stage) throws RebarException {
124         // delete protobuf contents of the stage:
125         for (byte prefix : stageCFPrefixes) {
126             Text cf = new Text(ByteUtil.fromIntWithPrefix(prefix, stage.getStageId()));
127             for (String tableName : protoTableNames)
128                 accumuloConnector.deleteColumnFamily(tableName, cf);
129         }
130         // delete record of the stage:
131         stageTable.deleteStage(stage);
132     }
