26  /** A collection backed by the set of all row identifiers in 
27   * a specified accumulo table.  This is used to special-case
28   * the task of reading all items from a corpus or knowledge
29   * graph. */
30  abstract public class AccumuloAllRowIdsCollection<Identifier>
31  	extends AbstractCollection<Identifier>
32  	implements Collection<Identifier> 
33  {
34  	// Hook for subclass:
35  	abstract protected Identifier rowIdToId(Text row);
36  
37  	// Private variables:
38  	private final AccumuloConnector accumuloConnector;
39  	private final String tableName;
40  	private Collection<Identifier> identifiers = null;
41  	
42  	public AccumuloAllRowIdsCollection(AccumuloConnector accumuloConnector, String tableName) {
43  		this.accumuloConnector = accumuloConnector;
44  		this.tableName = tableName;
45  	}
46  
47  	public Iterator<Identifier> iterator() {
48  		if (identifiers == null)
49  			identifiers = scanIdentifiers();
50  		return identifiers.iterator();
51  	}
52  
53  	public int size() {
54  		if (identifiers == null)
55  			identifiers = scanIdentifiers();
56  		return identifiers.size();
57  	}
58  
59  	private Collection<Identifier> scanIdentifiers() {
60  		try {
61  			identifiers = new HashSet<Identifier>();
62  			final Scanner scanner = accumuloConnector.createScanner(tableName);
63  			scanner.setRange(new Range());
64  			// [xx] the FirstEntryInRowIterator is broken in Accumulo 1.4 (it throws
65  			// a NullPointerException because a private variable does not get 
66  			// initialized in the constructor).  So leave this one out for now:
67  			//IteratorSetting is = new IteratorSetting(1000, //"first_entry",
68  			//                                         FirstEntryInRowIterator.class);
69  			scanner.addScanIterator(new IteratorSetting(1001, "key_only",
70  			SortedKeyIterator.class));
71  			final Iterator<Map.Entry<Key, Value>> iterator = scanner.iterator();
72  			while (iterator.hasNext()) {
73  				identifiers.add(rowIdToId(iterator.next().getKey().getRow()));
74  			}
75  			return identifiers;
76  		} catch (RebarException e) {
77  			throw new RuntimeException(e);
78  		}
79  	}
80  }
