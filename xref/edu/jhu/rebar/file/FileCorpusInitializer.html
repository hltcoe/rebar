28  public class FileCorpusInitializer
29              implements Corpus.Initializer {
30  
31      private final Path path;
32      private final Connection conn;
33      private final Set<String> existingGuidSet = new TreeSet<>();
34      private final Set<String> newGuidSet = new TreeSet<>();
35      
36      /**
37       * 
38       */
39      public FileCorpusInitializer(Path path, Connection conn) throws RebarException {
40          this.path = path;
41          this.conn = conn;
42          
43          try {
44              PreparedStatement ps = this.conn.prepareStatement("SELECT guid FROM comm_ids");
45              ResultSet rs = ps.executeQuery();
46              while (rs.next()) {
47                  this.existingGuidSet.add(rs.getString(1));
48              }
49              
50              rs.close();
51              ps.close();
52          } catch (SQLException e) {
53              throw new RebarException(e);
54          }
55      }
56  
57      @Override
58      public IndexedCommunication addCommunication(Communication comm) throws RebarException {
59          try {
60              String guid = comm.getGuid().getCommunicationId();
61              if (this.communicationExists(guid))
62                  throw new RebarException("Comm GUID: " + guid + " already exists in this corpus.");
63              this.newGuidSet.add(guid);
64              ProtocolBufferWriter pbw = new ProtocolBufferWriter(new FileOutputStream(this.path.toFile()));
65              pbw.write(comm);
66              pbw.close();
67              ProtoIndex pi = new ProtoIndex(comm);
68              return new IndexedCommunication(comm, pi, null);
69          } catch (IOException e) {
70              throw new RebarException(e);
71          } finally {
72              
73          }
74      }
75  
76      @Override
77      public boolean communicationExists(String commId) throws RebarException {
78          return this.newGuidSet.contains(commId) || this.existingGuidSet.contains(commId);
79      }
80  
81      @Override
82      public void close() throws RebarException {
83          try {
84              PreparedStatement ps = this.conn.prepareStatement("INSERT INTO comm_ids VALUES (?, ?)");
85              for (String s : this.newGuidSet) {
86                  ps.setString(1, null);
87                  ps.setString(2, s);
88                  ps.addBatch();
89              }
90              
91              ps.executeBatch();
92              ps.close();
93              
94              this.conn.close();
95          } catch (SQLException e) {
96              throw new RebarException(e);
97          }
98      }
