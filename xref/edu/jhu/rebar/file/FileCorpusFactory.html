85  		private void createSQLiteBackend (Connection conn) throws RebarException {
86  	    	try {
87  	    	Statement statement = conn.createStatement();
88  	    	statement.executeUpdate("CREATE TABLE stages (id integer PRIMARY KEY, name string, version string, description text)");
89  	    	statement.executeUpdate("CREATE TABLE dependencies (stage_id integer, dependency_id integer, " +
90  	    			"FOREIGN KEY(stage_id) REFERENCES stages(id))");
91  	    	statement.executeUpdate("CREATE TABLE comm_ids (id integer PRIMARY KEY, guid string)");
92  	    	statement.executeUpdate("CREATE TABLE '" + "stage_mods" + "' (stage_id integer, " +
93                          "comm_id string, " +
94                          "target blob, " +
95                          "mods blob, " +
96                          "FOREIGN KEY(stage_id) REFERENCES stages(id))");
97  	    	statement.close();
98  	        } catch (SQLException ioe) {
99  	            throw new RebarException(ioe);
100 	        }

103 		public FileBackedCorpus initialize() throws RebarException {
104 			try {
105 				this.createFoldersAndFiles();
106 				Connection conn = getConnection(this.name);
107 				this.createSQLiteBackend(conn);
108 				
109 				Set<String> commIdSet = new TreeSet<>();
110 				PreparedStatement ps = conn.prepareStatement("INSERT INTO comm_ids VALUES (?, ?)");
111 				while (this.commIter.hasNext()) {
112 					Communication comm = this.commIter.next();
113 					CommunicationGUID corpusNameGuid = CommunicationGUID.newBuilder(comm.getGuid())
114 							.setCorpusName(this.name)
115 							.build();
116 					comm = Communication.newBuilder(comm).setGuid(corpusNameGuid).build();
117 					
118 					String guid = comm.getGuid().getCommunicationId();
119 					boolean newComm = commIdSet.add(guid);
120 					if (newComm) {
121 						ps.setString(1, null);
122 						ps.setString(2, guid);
123 						ps.addBatch();
124 						
125 						this.writeCommunicationProtobufFile(comm);
126 					}
127 				}
128 				
129 				ps.executeBatch();
130 				
131 				corporaList.add(this.name);
132 				return new FileBackedCorpus(this.corpusPath, commIdSet, conn);
133 			} catch (SQLException e) {
134 				throw new RebarException(e);
135 			}
136 		}

167      */
168     public FileCorpusFactory(Path pathOnDisk) throws RebarException {
169         this.pathOnDisk = pathOnDisk;
170         this.corporaList = new TreeSet<>();
171         try {
172             if (!Files.exists(pathOnDisk)) {
173                 logger.info("Creating path: " + this.pathOnDisk.toString());
174                 Files.createDirectories(pathOnDisk);
175                 this.corporaList = new TreeSet<>();
176             } else {
177             	DirectoryStream<Path> ds = Files.newDirectoryStream(this.pathOnDisk);
178                 Iterator<Path> pathIter = ds.iterator();
179                 while (pathIter.hasNext())
180                     corporaList.add(pathIter.next().getFileName().toString());
181             }
182         } catch (IOException e) {
183             throw new RebarException(e);
184         }
185     }

190 			Iterator<Communication> commIdIter) throws RebarException {
191 		if (this.corpusExists(corpusName))
192 			throw new RebarException("Corpus " + corpusName + " already exists. Call getCorpus() instead.");
193 		FileCorpusFactoryInitializer init = new FileCorpusFactoryInitializer(corpusName, commIdIter);
194 		FileBackedCorpus fbc = init.initialize();
195 		return fbc;

231     	if (this.corpusExists(corpusName)) {
232     		Path p = this.getPathToCorpus(corpusName);
233     		return new FileBackedCorpus(p, this.getConnection(corpusName));
