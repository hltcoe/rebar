7   package edu.jhu.rebar;
8   
9   import java.util.Set;
10  import java.util.SortedSet;
11  
12  /**
13   * Base interface for corpora and graphs.
14   */
15  public interface StagedDataCollection {
16  
17      /** Return the name of this corpus. */
18      public String getName();
19  
20      /** Free any resources used by this corpus. */
21      public void close() throws RebarException;
22  
23      /** Return the total size of this corpus (in bytes). */
24      //public long getSize() throws RebarException;
25  
26      // ======================================================================
27      // Stages
28      // ======================================================================
29  
30      /**
31       * Create and return a new stage with the given name, version, and list of
32       * dependencies. The new stage will be assigned a new stageId that is unique
33       * within the corpus and is greater than the stageId of any existing stages.
34       * 
35       * @param dependencies
36       *            The stages that the new stage depends on.
37       * @param description
38       *            A string description of this stage. This should contain a
39       *            human-readable description of exactly what the stage adds.
40       * @param deleteIfExists
41       *            Controls behavior if a stage with the given name and version
42       *            already exists in this collection. If true, then delete the
43       *            existing stage before returning. If false, then raise an
44       *            exception.
45       * @raise RebarException if a stage with the given name and version already
46       *        exists (and deleteIfExists=false)
47       */
48      public Stage makeStage(String stageName, String stageVersion, Set<Stage> dependencies, String description, boolean deleteIfExists)
49              throws RebarException;
50  
51      /**
52       * Mark the given stage as "public." This should be done only after the
53       * stage has been fully written. For a stage to be marked public, all of its
54       * dependencies must first be marked public. Once a stage has been marked as
55       * public, it can not be made unpublic.
56       */
57      public void markStagePublic(Stage stage) throws RebarException;
58  
59      /**
60       * Look up and return the existing stage with the given name and version.
61       * 
62       * @raise RebarException if no such stage is found.
63       */
64      public Stage getStage(String stageName, String stageVersion) throws RebarException;
65  
66      /**
67       * Look up and return the existing stage with the given id.
68       * 
69       * @raise RebarException if no such stage is found.
70       */
71      public Stage getStage(int stageId) throws RebarException;
72  
73      /**
74       * Return the specified stage. If stageString has the form NNN:VVV, then NNN
75       * is treated as the name and VVV as the version. Otherwise, stageString is
76       * treated as a stage name, and the most recently created *public* stage
77       * with the given name is returned.
78       * 
79       * @raise RebarException if no such stage is found.
80       */
81      public Stage getStage(String stageString) throws RebarException;
82  
83      /**
84       * Return the set of all stages that have generated output for this corpus,
85       * sorted by their integer ids (i.e., by the order in which they were
86       * created).
87       */
88      public SortedSet<Stage> getStages() throws RebarException;
89  
90      /** Return true if a stage with the given name and version exists. */
91      public boolean hasStage(String stageName, String stageVersion) throws RebarException;
92  
93      /**
94       * Delete a specified stage. This will fail (and raise an exception) if any
95       * other stage depends on the given stage.
96       */
97      public void deleteStage(Stage stage) throws RebarException;
98  
99  }
