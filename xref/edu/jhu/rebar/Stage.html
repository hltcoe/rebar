9   import java.util.Set;

11  /**
12   * A record for a processing stage, which adds information to a REBAR corpus or
13   * knowledge graph. Each stage is uniquely identified (within a given corpus or
14   * knowledge graph) by a name string and a version string.
15   * 
16   * In addition, each processing stage is assigned a unique integer identifier
17   * when it is first created (unique within the corpus or knowledge graph that
18   * contains the stage). These are assigned in monotonic order, meaning that if
19   * stage Y was created after stage X, then Y's id will be greater than X's id.
20   * 
21   * Finally, each stage contains a list of stages that it depends on (i.e., that
22   * should be read before this stage when loading objects). The integer id of any
23   * stage is required to be strictly greater than the integer ids of any stages
24   * it depends on. (This ensures that there are no cycles in the dependency
25   * graph, and makes it easy to load the stages in the correct order.)
26   * 
27   * Stage objects are immutable. Equality and ordering comparisons between stages
28   * are based on their integer ids.
29   */
30  public interface Stage extends Comparable<Stage> {
31  
32      /** Return the name of this stage. */
33      public String getStageName();
34  
35      /** Return the version string for this stage. */
36      public String getStageVersion();
37  
38      /**
39       * Return the unique integer id for this stage. Stage ids are assigned in
40       * monotonic order.
41       */
42      public int getStageId();
43  
44      /**
45       * Return true if this stage has been made "public." Use
46       * StagedDataCollection::markStagePublic() to make a stage public.
47       */
48      public boolean isPublic();
49  
50      /**
51       * Return a list of the dependencies of this stage. The returned set is
52       * sorted (by stage id) and unmodifiable.
53       */
54      public Set<Stage> getDependencies();
55  
56      /** Return a description of what this stage contains. */
57      public String getDescription();
58  
59      /**
60       * Compares the specified object with this stage for equality. Two stage
61       * objects are considered equal if their integer ids are the same and they
62       * are owned by the same corpus or knowledge graph.
63       */
64      public boolean equals(Object o);
65  }
