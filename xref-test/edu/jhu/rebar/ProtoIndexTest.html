57      @Test
58      public void testAddSectionSegmentation() throws RebarException,
59              InvalidProtocolBufferException {
60          FieldDescriptor ssField = Concrete.Communication.getDescriptor()
61                  .findFieldByName("section_segmentation");
62          SectionSegmentation ssToAppend = ConcreteSectionSegmentation
63                  .generateSectionSegmentation(Tokenizer.TWITTER,
64                          commOne.getText());
65          this.pi.addField(this.pi.getRoot(), ssField, ssToAppend);
66  
67          Map<ModificationTarget, byte[]> modMap = this.pi
68                  .getUnsavedModifications();
69          for (Entry<ModificationTarget, byte[]> entry : modMap.entrySet()) {
70              ModificationTarget mt = entry.getKey();
71              assertEquals(commOne.getUuid(), mt.uuid);
72              byte[] mods = entry.getValue();
73              Communication mergedComm = commOne.toBuilder().mergeFrom(mods)
74                      .build();
75              assertEquals(commOne.toBuilder().addSectionSegmentation(ssToAppend)
76                      .build(), mergedComm);
77          }
78      }
